**Advanced Character Representation Analysis**  

The system performs multi-dimensional demographic analysis with intersectionality, using balanced sampling for ML fairness.

```python
from deepface import DeepFace
import pandas as pd
import numpy as np
from collections import Counter
from fairlearn.metrics import MetricFrame
from fairlearn.postprocessing import plot_threshold_optimizer
import matplotlib.pyplot as plt

class ComprehensiveDiversityAnalyzer:
    """
    Advanced demographic analysis system with intersectional bias detection.
    """

    def __init__(self):
        self.demographic_targets = {
            'gender': {'Woman': 45, 'Man': 45, 'Non-binary': 10},
            'age_groups': {
                'Teen (13-19)': 15,
                'Young Adult (20-35)': 35,
                'Middle Age (36-55)': 35,
                'Senior (55+)': 15
            },
            'ethnicity': {
                'White': 40,
                'Black/African American': 20,
                'Asian': 15,
                'Hispanic/Latino': 15,
                'Native American': 5,
                'Mixed/Other': 5
            }
        }

    def analyze_comprehensive_representation(self, character_data):
        """
        Perform multi-dimensional demographic analysis with intersectionality.
        """
        results = {
            'individual_demographics': {},
            'intersectional_analysis': {},
            'temporal_representation': {},
            'narrative_role_analysis': {},
            'bias_risk_assessment': {}
        }

        # Individual demographic analysis
        for attribute, targets in self.demographic_targets.items():
            actual_dist = Counter(character_data[attribute])
            total_chars = len(character_data)

            for group, target_pct in targets.items():
                actual_count = actual_dist.get(group, 0)
                actual_pct = actual_count / total_chars
                deviation = abs(actual_pct - target_pct)

                results['individual_demographics'][f"{attribute}_{group}"] = {
                    'actual_percentage': actual_pct * 100,
                    'target_percentage': target_pct * 100,
                    'deviation': deviation * 100,
                    'meets_threshold': deviation < 0.10  # 10% tolerance
                }

        # Intersectional analysis (gender + ethnicity)
        intersectional_data = []
        for char in character_data:
            intersectional_data.append(f"{char['gender']}_{char['ethnicity']}")

        intersectional_dist = Counter(intersectional_data)
        min_expected = len(character_data) * 0.02  # 2% minimum
        underrepresented = [
            {
                'intersection': intersection,
                'count': count,
                'percentage': (count / len(character_data)) * 100,
                'deficit': min_expected - count
            }
            for intersection, count in intersectional_dist.items() if count < min_expected
        ]

        results['intersectional_analysis'] = {
            'total_intersections': len(intersectional_dist),
            'underrepresented_groups': underrepresented,
            'diversity_index': self._calculate_diversity_index(intersectional_dist)
        }

        return results

    def analyze_narrative_role_bias(self, character_data):
        """
        Analyze demographics in narrative roles for pigeonholing.
        """
        role_demographics = {}
        for char in character_data:
            role = char['narrative_role']
            if role not in role_demographics:
                role_demographics[role] = {'gender': [], 'ethnicity': [], 'age': []}

            role_demographics[role]['gender'].append(char['gender'])
            role_demographics[role]['ethnicity'].append(char['ethnicity'])
            role_demographics[role]['age'].append(char['age'])

        bias_alerts = []
        for role, demographics in role_demographics.items():
            # Gender concentration check
            gender_dist = Counter(demographics['gender'])
            total_in_role = len(demographics['gender'])

            for gender, count in gender_dist.items():
                percentage = (count / total_in_role) * 100
                if percentage > 80:
                    bias_alerts.append({
                        'type': 'gender_concentration',
                        'role': role,
                        'demographic': gender,
                        'percentage': percentage,
                        'severity': 'high' if percentage > 90 else 'medium'
                    })

            # Ethnicity concentration check
            eth_dist = Counter(demographics['ethnicity'])
            for ethnicity, count in eth_dist.items():
                percentage = (count / total_in_role) * 100
                if percentage > 70:
                    bias_alerts.append({
                        'type': 'ethnicity_concentration',
                        'role': role,
                        'demographic': ethnicity,
                        'percentage': percentage,
                        'severity': 'high' if percentage > 85 else 'medium'
                    })

        return bias_alerts

    def _calculate_diversity_index(self, distribution):
        """
        Calculate normalized Shannon diversity index.
        """
        total = sum(distribution.values())
        if total <= 1:
            return 0

        shannon_index = 0
        for count in distribution.values():
            if count > 0:
                proportion = count / total
                shannon_index -= proportion * np.log(proportion)

        max_possible = np.log(len(distribution))
        return shannon_index / max_possible if max_possible > 0 else 0

# Usage example
analyzer = ComprehensiveDiversityAnalyzer()

# Sample character database (in practice, load from AI-generated data)
character_database = []  # Populate with 15 characters using DeepFace and script metadata
for i in range(15):
    char_image_path = f"characters/main_character_{i:02d}.jpg"
    analysis = DeepFace.analyze(char_image_path, actions=['age', 'gender', 'race'], enforce_detection=False)
    script_metadata = {}  # Load from script

    character_database.append({
        'character_id': i,
        'name': script_metadata.get('name', f'Character {i}'),
        'gender': analysis[0]['dominant_gender'],
        'age': analysis[0]['age'],
        'ethnicity': analysis[0]['dominant_race'],
        'narrative_role': script_metadata.get('role_type', 'supporting'),
        'screen_time_minutes': script_metadata.get('screen_time', 10),
        'dialogue_lines': script_metadata.get('dialogue_count', 50),
        'agency_score': script_metadata.get('character_agency', 5)
    })

diversity_results = analyzer.analyze_comprehensive_representation(character_database)
narrative_bias_results = analyzer.analyze_narrative_role_bias(character_database)

# Report generation
print("=" * 60)
print("COMPREHENSIVE CHARACTER DIVERSITY ANALYSIS")
print("=" * 60)

for demo_key, demo_data in diversity_results['individual_demographics'].items():
    status = "✓ PASS" if demo_data['meets_threshold'] else "⚠ NEEDS ATTENTION"
    print(f"{demo_key}: {demo_data['actual_percentage']:.1f}% "
          f"(target: {demo_data['target_percentage']:.1f}%) {status}")

print(f"\nIntersectional Diversity Index: {diversity_results['intersectional_analysis']['diversity_index']:.3f}")
print(f"Underrepresented Intersections: {len(diversity_results['intersectional_analysis']['underrepresented_groups'])}")

if narrative_bias_results:
    print("\n⚠ NARRATIVE ROLE BIAS DETECTED:")
    for bias in narrative_bias_results:
        print(f"  [{bias['severity'].upper()}] {bias['type']}: {bias['demographic']} "
              f"represents {bias['percentage']:.1f}% of {bias['role']} roles")
```

```python
                    'category': category,
                    'confidence': confidence,
                    'threshold': adjusted_threshold,
                    'excess': confidence - adjusted_threshold,
                    'severity': processed_detection['severity']
                })

        analysis['risk_assessment'] = self._calculate_image_risk_assessment(
            analysis['processed_detections']
        )

        return analysis

    def _calculate_detection_severity(self, category, confidence, threshold):
        """
        Calculate severity based on category and confidence.
        """
        category_severities = {
            'FEMALE_GENITALIA_EXPOSED': 10,
            'MALE_GENITALIA_EXPOSED': 10,
            'ANUS_EXPOSED': 10,
            'BUTTOCKS_EXPOSED': 7,
            'FEMALE_BREAST_EXPOSED': 6,
            'MALE_BREAST_EXPOSED': 2,
            'BELLY_EXPOSED': 1,
            'FEET_EXPOSED': 1,
            'ARMPITS_EXPOSED': 1
        }

        base_severity = category_severities.get(category, 1)

        if confidence <= threshold:
            return 0

        excess_factor = (confidence - threshold) / (1 - threshold) if threshold < 1 else 0
        final_severity = int(base_severity * excess_factor)
        return min(final_severity, 10)

    def _calculate_image_risk_assessment(self, processed_detections):
        """
        Calculate comprehensive risk assessment for an image.
        """
        if not processed_detections:
            return {'risk_level': 'safe', 'risk_score': 0}

        violation_count = sum(1 for d in processed_detections if d['violates_rating'])
        max_severity = max((d['severity'] for d in processed_detections), default=0)
        avg_confidence = np.mean([d['confidence'] for d in processed_detections])

        total_bbox_area = sum(d['bbox_area'] for d in processed_detections)
        image_area = 1024 * 1024  # Default; calculate from actual image
        exposure_percentage = (total_bbox_area / image_area) * 100

        risk_components = {
            'violation_penalty': violation_count * 20,
            'severity_penalty': max_severity * 5,
            'confidence_penalty': avg_confidence * 10,
            'exposure_penalty': min(exposure_percentage, 30)
        }

        total_risk_score = sum(risk_components.values())
        normalized_risk_score = min(total_risk_score / 100, 1)

        if normalized_risk_score == 0:
            risk_level = 'safe'
        elif normalized_risk_score <= 0.3:
            risk_level = 'low'
        elif normalized_risk_score <= 0.6:
            risk_level = 'moderate'
        elif normalized_risk_score <= 0.8:
            risk_level = 'high'
        else:
            risk_level = 'critical'

        return {
            'risk_level': risk_level,
            'risk_score': normalized_risk_score,
            'risk_components': risk_components,
            'violation_count': violation_count,
            'max_severity': max_severity,
            'exposure_percentage': exposure_percentage
        }

    def generate_censored_versions(self, violation_analyses, censoring_method='adaptive_blur'):
        """
        Generate censored versions of images with NSFW violations.
        """
        censored_results = []

        for analysis in violation_analyses:
            if not analysis['violations']:
                continue

            image_path = analysis['image_path']
            image = cv2.imread(image_path)
            if image is None:
                continue

            censored_image = image.copy()

            for detection in analysis['processed_detections']:
                if detection['violates_rating']:
                    bbox = detection['bbox']
                    x, y, w, h = bbox
                    region = censored_image[y:y+h, x:x+w]

                    if censoring_method == 'adaptive_blur':
                        blur_strength = min(detection['severity'] * 3, 30)
                        if blur_strength > 0:
                            blurred_region = cv2.GaussianBlur(region, (blur_strength | 1, blur_strength | 1), 0)
                            censored_image[y:y+h, x:x+w] = blurred_region

                    elif censoring_method == 'pixelate':
                        pixel_size = min(detection['severity'] * 2, 20)
                        if pixel_size > 1:
                            small = cv2.resize(region, (w // pixel_size, h // pixel_size), interpolation=cv2.INTER_LINEAR)
                            pixelated = cv2.resize(small, (w, h), interpolation=cv2.INTER_NEAREST)
                            censored_image[y:y+h, x:x+w] = pixelated

                    elif censoring_method == 'black_bar':
                        cv2.rectangle(censored_image, (x, y), (x + w, y + h), (0, 0, 0), -1)

            censored_path = image_path.replace('.jpg', '_censored.jpg')
            cv2.imwrite(censored_path, censored_image)

            censored_results.append({
                'original_path': image_path,
                'censored_path': censored_path,
                'censoring_method': censoring_method,
                'regions_censored': len([d for d in analysis['processed_detections'] if d['violates_rating']])
            })

        return censored_results

        }

    def _check_rating_compliance(self, image_analyses, target_rating):
```

### Primary Framework: Microsoft FairLearn

FairLearn is an open-source toolkit for assessing and improving AI fairness, compatible with Python 3.9+ and ML frameworks like scikit-learn, TensorFlow, and PyTorch.

**Implementation Approach**

```python
from fairlearn.metrics import MetricFrame, demographic_parity_difference, equalized_odds_difference
from fairlearn.reductions import ExponentiatedGradient, DemographicParity
from fairlearn.postprocessing import ThresholdOptimizer

class EthicsBiasDetector:
    def __init__(self):
        self.fairness_metrics = {
            'demographic_parity': demographic_parity_difference,
            'equalized_odds': equalized_odds_difference
        }

    def assess_model_fairness(self, y_true, y_pred, sensitive_features):
        """
        Assess model fairness across sensitive attributes.
        """
        metric_frame = MetricFrame(
            metrics=self.fairness_metrics,
            y_true=y_true,
            y_pred=y_pred,
            sensitive_features=sensitive_features
        )
        return metric_frame.by_group

    def mitigate_bias(self, estimator, X_train, y_train, sensitive_features):
        """
        Mitigate bias using exponentiated gradient reduction.
        """
        mitigator = ExponentiatedGradient(
            estimator=estimator,
            constraints=DemographicParity()
        )
        mitigator.fit(X_train, y_train, sensitive_features=sensitive_features)
        return mitigator.predict(X_train), mitigator.predict_proba(X_train)
```

```python
from azure.ai.contentsafety import ContentSafetyClient
from azure.core.credentials import AzureKeyCredential
from azure.ai.contentsafety.models import AnalyzeTextOptions

class ContentSafetyAnalyzer:
    def __init__(self, endpoint, api_key):
        self.client = ContentSafetyClient(endpoint, AzureKeyCredential(api_key))
        self.severity_thresholds = {
            'G': {'violence': 0, 'hate': 0, 'sexual': 0, 'self_harm': 0},
            'PG': {'violence': 2, 'hate': 2, 'sexual': 0, 'self_harm': 0},
            'PG-13': {'violence': 4, 'hate': 4, 'sexual': 2, 'self_harm': 2},
            'R': {'violence': 6, 'hate': 6, 'sexual': 4, 'self_harm': 4}
        }

    def analyze_content(self, content, content_type='text', target_rating='PG-13'):
        """
        Analyze content for safety violations.
        """
        if content_type == 'text':
            poller = self.client.analyze_text(
                AnalyzeTextOptions(text=content),
                content_filter="Default"
            )
            result = poller.result()
        else:
            # Image analysis implementation
            result = None  # Placeholder

        violations = self._check_violations(result, target_rating)
        return {
            'compliant': len(violations) == 0,
            'violations': violations,
            'detailed_scores': result
        }

    def _check_violations(self, result, target_rating):
        """
        Check analysis results against rating thresholds.
        """
        violations = []
        thresholds = self.severity_thresholds[target_rating]

        for category in ['Violence', 'Hate', 'Sexual', 'SelfHarm']:
            cat_result = getattr(result, f"{category.lower()}_result", None)
            if cat_result and cat_result.severity > thresholds.get(category.lower(), 0):
                violations.append({
                    'category': category,
                    'severity': cat_result.severity,
                    'threshold': thresholds.get(category.lower(), 0)
                })

        return violations
```

```python
from nudenet import NudeDetector

class NSFWContentAnalyzer:
    def __init__(self, model_path=None, inference_resolution=320):
        self.detector = NudeDetector(model_path=model_path, inference_resolution=inference_resolution)
        self.nsfw_categories = [
            'BUTTOCKS_EXPOSED', 'FEMALE_BREAST_EXPOSED', 
            'FEMALE_GENITALIA_EXPOSED', 'MALE_GENITALIA_EXPOSED', 
            'ANUS_EXPOSED'
        ]
        self.rating_thresholds = {
            'G': 0.0, 'PG': 0.1, 'PG-13': 0.3, 'R': 0.6
        }

    def analyze_image_batch(self, image_paths, target_rating='PG-13'):
        """
        Analyze batch of images for NSFW content.
        """
        results = []
        detections = self.detector.detect_batch(image_paths)

        for image_path, detection_result in zip(image_paths, detections):
            nsfw_score = self._calculate_nsfw_score(detection_result)
            threshold = self.rating_thresholds[target_rating]

            result = {
                'image_path': image_path,
                'nsfw_score': nsfw_score,
                'compliant': nsfw_score <= threshold,
                'detections': detection_result,
                'censored_version': None
            }

            if nsfw_score > threshold:
                censored_path = self.detector.censor(
                    image_path, 
                    classes=self.nsfw_categories,
                    output_path=image_path.replace('.jpg', '_censored.jpg')
                )
                result['censored_version'] = censored_path

            results.append(result)

        return results

    def _calculate_nsfw_score(self, detections):
        """
        Calculate overall NSFW score from detections.
        """
        max_score = 0
        for detection in detections:
            if detection['class'] in self.nsfw_categories:
                max_score = max(max_score, detection['score'])
        return max_score
```
