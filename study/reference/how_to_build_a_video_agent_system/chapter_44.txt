```python
class AdvancedDataQualityAnalyzer:
    """

    """

    def __init__(self, min_resolution=(224, 224), quality_threshold=7.0):
        self.min_resolution = min_resolution
        self.quality_threshold = quality_threshold
        self.logger = self._setup_logging()

        # Initialize quality assessment models
        self.blur_detector = self._load_blur_detector()

    def _setup_logging(self):

        logger = logging.getLogger('DataQualityAnalyzer')
        logger.setLevel(logging.INFO)

        handler = logging.FileHandler('data_quality_analysis.log')

        )
        handler.setFormatter(formatter)
        logger.addHandler(handler)

        return logger

    def comprehensive_quality_assessment(self, image_path):
        """
        Multi-dimensional quality assessment including:
        - Technical quality (blur, noise, exposure)
        - Content quality (face detection confidence, occlusion analysis)
        - Metadata analysis (EXIF data, source reliability)
        """
        try:
            image = cv2.imread(image_path)
            if image is None:
                return {"valid": False, "reason": "Failed to load image"}

            # Convert to RGB for consistency
            image_rgb = cv2.cvtColor(image, cv2.COLOR_BGR2RGB)

            quality_metrics = {}

            # Resolution Analysis
            height, width = image.shape[:2]
            quality_metrics['resolution'] = {
                'width': width,
                'height': height,
                'meets_minimum': width >= self.min_resolution[0] and height >= self.min_resolution[1],
                'aspect_ratio': width / height,
                'total_pixels': width * height
            }

            # Sharpness Assessment (Multiple Methods)
            quality_metrics['sharpness'] = self._assess_sharpness(image)

            # Exposure and Dynamic Range Analysis
            quality_metrics['exposure'] = self._analyze_exposure(image_rgb)

            # Noise Level Assessment
            quality_metrics['noise'] = self._assess_noise_level(image)

            # Color Quality Analysis
            quality_metrics['color'] = self._analyze_color_quality(image_rgb)

            # Face-specific Quality Metrics (if applicable)
            quality_metrics['face_quality'] = self._assess_face_quality(image_rgb)

            # Overall Quality Score Calculation
            overall_score = self._calculate_overall_score(quality_metrics)
            quality_metrics['overall_score'] = overall_score
            quality_metrics['valid'] = overall_score >= self.quality_threshold

            # Log detailed analysis
            self.logger.info(f"Quality analysis for {image_path}: Score={overall_score:.3f}")

            return quality_metrics

        except Exception as e:
            self.logger.error(f"Error analyzing {image_path}: {str(e)}")
            return {"valid": False, "reason": f"Analysis error: {str(e)}"}

    def _assess_sharpness(self, image):
        """Multi-method sharpness assessment for robust evaluation"""
        gray = cv2.cvtColor(image, cv2.COLOR_BGR2GRAY)

        # Method 1: Laplacian Variance (traditional)
        laplacian_var = cv2.Laplacian(gray, cv2.CV_64F).var()

        # Method 2: Sobel Edge Detection
        sobel_x = cv2.Sobel(gray, cv2.CV_64F, 1, 0, ksize=3)
        sobel_y = cv2.Sobel(gray, cv2.CV_64F, 0, 1, ksize=3)
        sobel_magnitude = np.sqrt(sobel_x**2 + sobel_y**2)
        sobel_mean = np.mean(sobel_magnitude)

        # Method 3: High-frequency content analysis
        fft = np.fft.fft2(gray)
        fshift = np.fft.fftshift(fft)
        magnitude_spectrum = np.log(np.abs(fshift) + 1)

        # Calculate high-frequency energy
        h, w = magnitude_spectrum.shape
        center_h, center_w = h // 2, w // 2
        high_freq_region = magnitude_spectrum.copy()
        high_freq_region[center_h-30:center_h+30, center_w-30:center_w+30] = 0
        high_freq_energy = np.mean(high_freq_region)

        return {
            'laplacian_variance': float(laplacian_var),
            'sobel_mean': float(sobel_mean),
            'high_freq_energy': float(high_freq_energy),
            'is_sharp': laplacian_var > 100,  # Threshold can be adjusted
            'sharpness_score': min(10.0, laplacian_var / 500)  # Normalized score
        }

    def _analyze_exposure(self, image_rgb):
        """Comprehensive exposure and dynamic range analysis"""
        # Convert to different color spaces for analysis
        hsv = cv2.cvtColor(image_rgb, cv2.COLOR_RGB2HSV)
        lab = cv2.cvtColor(image_rgb, cv2.COLOR_RGB2LAB)

        # Luminance analysis
        luminance = lab[:, :, 0]  # L channel in LAB

        # Calculate exposure metrics
        mean_luminance = np.mean(luminance)
        std_luminance = np.std(luminance)

        # Check for clipping
        highlight_clipping = np.sum(luminance > 245) / luminance.size
        shadow_clipping = np.sum(luminance < 10) / luminance.size

        # Dynamic range assessment
        dynamic_range = np.percentile(luminance, 99) - np.percentile(luminance, 1)

        # Histogram analysis
        hist, _ = np.histogram(luminance, bins=256, range=(0, 255))

        return {
            'mean_luminance': float(mean_luminance),
            'std_luminance': float(std_luminance),
            'highlight_clipping_ratio': float(highlight_clipping),
            'shadow_clipping_ratio': float(shadow_clipping),
            'dynamic_range': float(dynamic_range),
            'is_well_exposed': (
                0.02 <= highlight_clipping <= 0.02 and
                0.02 <= shadow_clipping <= 0.02 and
                dynamic_range > 100
            ),
            'exposure_score': self._calculate_exposure_score(
                mean_luminance, highlight_clipping, shadow_clipping, dynamic_range
            )
        }

    def _assess_noise_level(self, image):
        """Advanced noise assessment using multiple techniques"""
        gray = cv2.cvtColor(image, cv2.COLOR_BGR2GRAY)

        # Method 1: Standard deviation in uniform regions
        # Apply Gaussian blur and compare with original
        blurred = cv2.GaussianBlur(gray, (5, 5), 0)
        noise_map = cv2.absdiff(gray, blurred)
        noise_std = np.std(noise_map)

        # Method 2: High-frequency noise analysis
        # Apply median filter and analyze difference
        median_filtered = cv2.medianBlur(gray, 5)
        high_freq_noise = cv2.absdiff(gray, median_filtered)
        hf_noise_mean = np.mean(high_freq_noise)

        # Method 3: Local variance analysis
        kernel = np.ones((5, 5), np.float32) / 25
        local_mean = cv2.filter2D(gray.astype(np.float32), -1, kernel)
        local_variance = cv2.filter2D((gray.astype(np.float32) - local_mean)**2, -1, kernel)
        variance_std = np.std(local_variance)

        return {
            'noise_std': float(noise_std),
            'high_freq_noise_mean': float(hf_noise_mean),
            'variance_std': float(variance_std),
            'is_low_noise': noise_std < 15,  # Adjustable threshold
            'noise_score': max(0.0, 1.0 - (noise_std / 50))  # Normalized score
        }

    def _analyze_color_quality(self, image_rgb):
        """Comprehensive color quality assessment"""
        # Color space analysis
        hsv = cv2.cvtColor(image_rgb, cv2.COLOR_RGB2HSV)

        # Saturation analysis
        saturation = hsv[:, :, 1]
        mean_saturation = np.mean(saturation)
        saturation_std = np.std(saturation)

        # Color cast detection (using gray world assumption)
        r_mean, g_mean, b_mean = np.mean(image_rgb, axis=(0, 1))
        color_cast = np.std([r_mean, g_mean, b_mean])

        # Color histogram analysis
        color_histograms = []
        for i in range(3):  # RGB channels
            hist, _ = np.histogram(image_rgb[:, :, i], bins=256, range=(0, 255))
            color_histograms.append(hist)

        # Calculate color distribution entropy
        color_entropy = []
        for hist in color_histograms:
            normalized_hist = hist / np.sum(hist)
            # Avoid log(0) by adding small epsilon
            entropy = -np.sum(normalized_hist * np.log(normalized_hist + 1e-10))
            color_entropy.append(entropy)

        return {
            'mean_saturation': float(mean_saturation),
            'saturation_std': float(saturation_std),
            'color_cast': float(color_cast),
            'rgb_entropy': [float(e) for e in color_entropy],
            'mean_entropy': float(np.mean(color_entropy)),
            'is_vibrant': mean_saturation > 50 and np.mean(color_entropy) > 4.0,
            'color_score': min(10.0, (mean_saturation / 255) * (np.mean(color_entropy) / 8.0))
        }

    def _assess_face_quality(self, image_rgb):
        """Face-specific quality metrics using advanced detection"""
        try:
            # Use more sophisticated face detection
            face_cascade = cv2.CascadeClassifier(
                cv2.data.haarcascades + 'haarcascade_frontalface_default.xml'
            )
            gray = cv2.cvtColor(image_rgb, cv2.COLOR_RGB2GRAY)

            # Multiple scale detection for better accuracy
            faces = face_cascade.detectMultiScale(
                gray, 
                scaleFactor=1.1, 
                minNeighbors=5,
                minSize=(50, 50),
                maxSize=(500, 500)
            )

            if len(faces) == 0:
                return {
                    'faces_detected': 0,
                    'face_confidence': 0.0,
                    'largest_face_ratio': 0.0,
                    'face_quality_score': 0.0
                }

            # Analyze the largest/most confident face
            largest_face = max(faces, key=lambda f: f[2] * f[3])  # width * height
            x, y, w, h = largest_face

            # Face size ratio to image
            image_area = image_rgb.shape[0] * image_rgb.shape[1]
            face_area = w * h
            face_ratio = face_area / image_area

            # Face region quality analysis
            face_region = image_rgb[y:y+h, x:x+w]
            face_sharpness = self._assess_sharpness(
                cv2.cvtColor(face_region, cv2.COLOR_RGB2BGR)
            )

            # Eye detection within face region for quality assessment
            eye_cascade = cv2.CascadeClassifier(
                cv2.data.haarcascades + 'haarcascade_eye.xml'
            )
            face_gray = gray[y:y+h, x:x+w]
            eyes = eye_cascade.detectMultiScale(face_gray, 1.1, 5)

            return {
                'faces_detected': len(faces),
                'face_confidence': float(len(faces)),  # Simple confidence measure
                'largest_face_ratio': float(face_ratio),
                'eyes_detected': len(eyes),
                'face_sharpness_score': face_sharpness['sharpness_score'],
                'face_quality_score': min(10.0, face_ratio * 20) * face_sharpness['sharpness_score']
            }

        except Exception as e:
            return {
                'faces_detected': 0,
                'face_confidence': 0.0,
                'largest_face_ratio': 0.0,
                'face_quality_score': 0.0,
                'error': str(e)
            }

    def _calculate_overall_score(self, quality_metrics):
        """Calculate weighted overall quality score"""
        weights = {
            'sharpness': 0.3,
            'exposure': 0.2,
            'noise': 0.2,
            'color': 0.15,
            'face_quality': 0.15
        }
        
        total_score = 0.0
        total_weight = 0.0
        
        for metric, weight in weights.items():
            if metric in quality_metrics:
                score = quality_metrics[metric].get('sharpness_score' if metric == 'sharpness' 
                                                  else f'{metric}_score', 0.0)
                total_score += score * weight
                total_weight += weight
        
        return total_score / total_weight if total_weight > 0 else 0.0
```

```python
import optuna
from optuna.samplers import TPESampler, NSGAIISampler

@dataclass
class OptimizationConfig:
    """Configuration for hyperparameter optimization"""
    n_trials: int = 100
    timeout: int = 3600  # 1 hour
    n_jobs: int = 1
    study_name: str = "model_optimization"
    storage_url: Optional[str] = None
    load_if_exists: bool = True

    # Multi-objective settings
    directions: List[str] = None  # ["minimize", "maximize"] for loss and accuracy

    # Pruning settings
    pruner_type: str = "median"  # "median", "hyperband", "none"
    pruner_config: Dict = None

    # Sampler settings
    sampler_type: str = "tpe"  # "tpe", "nsga2", "random"
    sampler_config: Dict = None

class AdvancedHyperparameterOptimizer:
    """
    Comprehensive hyperparameter optimization system with multi-objective
    optimization, advanced pruning, and distributed computing support
    """

    def __init__(self, config: OptimizationConfig):
        self.config = config
        self.study = None
        self.best_trials_history = []

        # Initialize MLflow for experiment tracking
        mlflow.set_tracking_uri("sqlite:///mlruns.db")
        mlflow.set_experiment("hyperparameter_optimization")

    def create_study(self):
        """Create Optuna study with specified configuration"""
        # Set up pruner
        pruner = self._create_pruner()

        # Set up sampler
        sampler = self._create_sampler()

        # Set up directions for multi-objective optimization
        directions = self.config.directions or ["minimize"]

        # Create study
        self.study = optuna.create_study(
            study_name=self.config.study_name,
            storage=self.config.storage_url,
            load_if_exists=self.config.load_if_exists,
            directions=directions,
            pruner=pruner,
            sampler=sampler
        )

        return self.study

    def _create_pruner(self):
        """Create pruner based on configuration"""
        if self.config.pruner_type == "median":
            config = self.config.pruner_config or {}
            return MedianPruner(
                n_startup_trials=config.get("n_startup_trials", 5),
                n_warmup_steps=config.get("n_warmup_steps", 10),
                interval_steps=config.get("interval_steps", 1)
            )
        elif self.config.pruner_type == "hyperband":
            config = self.config.pruner_config or {}
            return HyperbandPruner(
                min_resource=config.get("min_resource", 1),
                max_resource=config.get("max_resource", 100),
                reduction_factor=config.get("reduction_factor", 3)
            )
        else:
            return optuna.pruners.NopPruner()

    def _create_sampler(self):
        """Create sampler based on configuration"""
        if self.config.sampler_type == "tpe":
            config = self.config.sampler_config or {}
            return TPESampler(
                n_startup_trials=config.get("n_startup_trials", 10),
                n_ei_candidates=config.get("n_ei_candidates", 24),
                seed=config.get("seed", 42)
            )
        elif self.config.sampler_type == "nsga2":
            config = self.config.sampler_config or {}
            return NSGAIISampler(
                population_size=config.get("population_size", 50),
                mutation_prob=config.get("mutation_prob", 0.1),
                crossover_prob=config.get("crossover_prob", 0.9),
                seed=config.get("seed", 42)
            )
        else:
            return optuna.samplers.RandomSampler(seed=42)

    def define_search_space(self, trial: optuna.Trial) -> Dict:
        """
        Define comprehensive search space for model hyperparameters
        """
        search_space = {}

        # Architecture parameters
        search_space['model_architecture'] = trial.suggest_categorical(
            'model_architecture', 
            ['resnet18', 'resnet34', 'resnet50', 'efficientnet_b0', 'efficientnet_b1']
        )

        # Training parameters
        search_space['learning_rate'] = trial.suggest_float(
            'learning_rate', 1e-5, 1e-1, log=True
        )

        search_space['batch_size'] = trial.suggest_categorical(
            'batch_size', [16, 32, 64, 128]
        )

        search_space['optimizer'] = trial.suggest_categorical(
            'optimizer', ['adam', 'sgd', 'adamw', 'rmsprop']
        )

        # Optimizer-specific parameters
        if search_space['optimizer'] == 'sgd':
            search_space['momentum'] = trial.suggest_float('momentum', 0.5, 0.99)
            search_space['weight_decay'] = trial.suggest_float(
                'weight_decay', 1e-6, 1e-2, log=True
            )
        elif search_space['optimizer'] in ['adam', 'adamw']:
            search_space['beta1'] = trial.suggest_float('beta1', 0.8, 0.99)
            search_space['beta2'] = trial.suggest_float('beta2', 0.9, 0.999)
            search_space['weight_decay'] = trial.suggest_float(
                'weight_decay', 1e-6, 1e-2, log=True
            )

        # Regularization parameters
        search_space['dropout_rate'] = trial.suggest_float('dropout_rate', 0.1, 0.7)
        search_space['label_smoothing'] = trial.suggest_float('label_smoothing', 0.0, 0.3)

        # Data augmentation parameters
        search_space['augmentation_strength'] = trial.suggest_float(
            'augmentation_strength', 0.1, 1.0
        )
        search_space['mixup_alpha'] = trial.suggest_float('mixup_alpha', 0.0, 1.0)
        search_space['cutmix_alpha'] = trial.suggest_float('cutmix_alpha', 0.0, 1.0)

        # Learning rate scheduling
        search_space['lr_scheduler'] = trial.suggest_categorical(
            'lr_scheduler', ['cosine', 'step', 'exponential', 'reduce_on_plateau']
        )

        if search_space['lr_scheduler'] == 'step':
            search_space['step_size'] = trial.suggest_int('step_size', 10, 50)
            search_space['gamma'] = trial.suggest_float('gamma', 0.1, 0.9)
        elif search_space['lr_scheduler'] == 'exponential':
            search_space['gamma'] = trial.suggest_float('gamma', 0.9, 0.99)

        # Model-specific parameters
        if 'resnet' in search_space['model_architecture']:
            search_space['pretrained'] = trial.suggest_categorical('pretrained', [True, False])

        # Training strategy
        search_space['warmup_epochs'] = trial.suggest_int('warmup_epochs', 0, 10)
        search_space['max_epochs'] = trial.suggest_int('max_epochs', 20, 100)

        return search_space

    def objective_function(self, trial: optuna.Trial, 
                          train_loader: DataLoader,
                          val_loader: DataLoader,
                          device: torch.device) -> Tuple[float, float]:
        """
        Multi-objective optimization function that returns both loss and accuracy
        """
        with mlflow.start_run(nested=True):
            # Get hyperparameters for this trial
            params = self.define_search_space(trial)

            # Log parameters to MLflow
            mlflow.log_params(params)

            # Create model based on hyperparameters
            model = self._create_model(params).to(device)

            # Create optimizer
            optimizer = self._create_optimizer(model, params)

            # Create loss function
            criterion = self._create_criterion(params)

            # Create learning rate scheduler
            scheduler = self._create_scheduler(optimizer, params)

            # Training loop with early stopping and pruning
            best_val_loss = float('inf')
            best_val_accuracy = 0.0
            patience_counter = 0
            max_patience = 10

            for epoch in range(params['max_epochs']):
                # Training phase
                model.train()
                train_loss = 0.0
                train_correct = 0
                train_total = 0

                for batch_idx, (images, labels) in enumerate(train_loader):
                    images, labels = images.to(device), labels.to(device)

                    optimizer.zero_grad()
                    outputs = model(images)
                    loss = criterion(outputs, labels)
                    loss.backward()
                    optimizer.step()

                    train_loss += loss.item()
                    _, predicted = torch.max(outputs.data, 1)
                    train_total += labels.size(0)
                    train_correct += (predicted == labels).sum().item()

                # Validation phase
                model.eval()
                val_loss = 0.0
                val_correct = 0
                val_total = 0

                with torch.no_grad():
                    for images, labels in val_loader:
                        images, labels = images.to(device), labels.to(device)
                        outputs = model(images)
                        loss = criterion(outputs, labels)

                        val_loss += loss.item()
                        _, predicted = torch.max(outputs.data, 1)
                        val_total += labels.size(0)
                        val_correct += (predicted == labels).sum().item()

                # Calculate metrics
                avg_train_loss = train_loss / len(train_loader)
                avg_val_loss = val_loss / len(val_loader)
                train_accuracy = train_correct / train_total
                val_accuracy = val_correct / val_total

                # Log metrics to MLflow
                mlflow.log_metrics({
                    'train_loss': avg_train_loss,
                    'val_loss': avg_val_loss,
                    'train_accuracy': train_accuracy,
                    'val_accuracy': val_accuracy
                }, step=epoch)

                # Report intermediate values to Optuna for pruning
                trial.report(avg_val_loss, epoch)
                trial.report(val_accuracy, epoch + 0.5)  # Offset for multi-objective

                # Check if trial should be pruned
                if trial.should_prune():
                    raise optuna.TrialPruned()

                # Early stopping logic
                if val_accuracy > best_val_accuracy:
                    best_val_accuracy = val_accuracy
                    best_val_loss = avg_val_loss
                    patience_counter = 0

                    # Save best model
                    torch.save(model.state_dict(), f'best_model_trial_{trial.number}.pth')
                else:
                    patience_counter += 1

                if patience_counter >= max_patience:
                    print(f"Early stopping at epoch {epoch}")
                    break

                # Update learning rate
                if scheduler:
                    if params['lr_scheduler'] == 'reduce_on_plateau':
                        scheduler.step(avg_val_loss)
                    else:
                        scheduler.step()

            # Clean up model to save memory
            del model
            torch.cuda.empty_cache()

            # Return objectives for multi-objective optimization
            return best_val_loss, best_val_accuracy

    def _create_model(self, params: Dict) -> nn.Module:
        """Create model based on hyperparameters"""
        # Implementation for model creation
        from torchvision import models
        if params['model_architecture'] == 'resnet50':
            model = models.resnet50(pretrained=params.get('pretrained', True))
            num_ftrs = model.fc.in_features
            model.fc = nn.Linear(num_ftrs, 10)  # Adjust for your task
        else:
            raise ValueError(f"Unsupported architecture: {params['model_architecture']}")
        return model

    def _create_optimizer(self, model: nn.Module, params: Dict):
        """Create optimizer based on hyperparameters"""
        if params['optimizer'] == 'adam':
            return torch.optim.Adam(model.parameters(), 
                                  lr=params['learning_rate'],
                                  weight_decay=params.get('weight_decay', 1e-4))
        elif params['optimizer'] == 'sgd':
            return torch.optim.SGD(model.parameters(),
                                 lr=params['learning_rate'],
                                 momentum=params.get('momentum', 0.9),
                                 weight_decay=params.get('weight_decay', 1e-4))
        else:
            raise ValueError(f"Unsupported optimizer: {params['optimizer']}")

    def _create_criterion(self, params: Dict):
        """Create loss function with optional label smoothing"""
        if params.get('label_smoothing', 0.0) > 0:
            # Implementation for label smoothing
            return nn.CrossEntropyLoss()
        return nn.CrossEntropyLoss()

    def _create_scheduler(self, optimizer, params: Dict):
        """Create learning rate scheduler"""
        scheduler_type = params.get('lr_scheduler', 'cosine')
        if scheduler_type == 'cosine':
            return torch.optim.lr_scheduler.CosineAnnealingLR(optimizer, T_max=50)
        elif scheduler_type == 'step':
            return torch.optim.lr_scheduler.StepLR(
                optimizer, 
                step_size=params.get('step_size', 30),
                gamma=params.get('gamma', 0.1)
            )
        return None
```

```python
class AdvancedClassifierHead(nn.Module):
    """
    Sophisticated classifier head with attention mechanism
    and advanced regularization techniques
    """

    def __init__(self, 
                 input_features: int, 
                 num_classes: int,
                 use_attention: bool = True,
                 dropout_rate: float = 0.5):
        super().__init__()

        self.use_attention = use_attention

        # Feature transformation
        self.feature_transform = nn.Sequential(
            nn.Linear(input_features, 1024),
            nn.BatchNorm1d(1024),
            nn.ReLU(),
            nn.Dropout(dropout_rate)
        )

        # Attention mechanism
        if use_attention:
            self.attention = SpatialAttentionModule(1024)

        # Classification layers
        self.classifier = nn.Sequential(
            nn.Linear(1024, 512),
            nn.BatchNorm1d(512),
            nn.ReLU(),
            nn.Dropout(dropout_rate * 0.5),
            nn.Linear(512, num_classes)
        )

        # Initialize weights
        self._initialize_weights()

    def forward(self, x):
        # Feature transformation
        features = self.feature_transform(x)

        # Apply attention if enabled
        if self.use_attention:
            # Reshape for attention (assuming features from CNN)
            # This would need adaptation based on actual input
            attended_features = self.attention(features)
            features = features * attended_features

        # Classification
        output = self.classifier(features)

        return output

    def _initialize_weights(self):
        """Initialize weights using Xavier/He initialization"""
        for m in self.modules():
            if isinstance(m, nn.Linear):
                nn.init.kaiming_normal_(m.weight, mode='fan_out', nonlinearity='relu')
                if m.bias is not None:
                    nn.init.constant_(m.bias, 0)
            elif isinstance(m, nn.BatchNorm1d):
                nn.init.constant_(m.weight, 1)
                nn.init.constant_(m.bias, 0)


class SpatialAttentionModule(nn.Module):

    """

    def __init__(self, features: int):

            nn.Linear(features, features // 8),
            nn.ReLU(),
            nn.Linear(features // 8, features),
            nn.Sigmoid()
        )

    def forward(self, x):
```

```python
    def analyze_gradients(self, model: torch.nn.Module) -> Dict:
        """Analyze gradient statistics"""

        param_count = 0
        gradient_stats = {}

        for name, param in model.named_parameters():
            if param.grad is not None:
                param_norm = param.grad.data.norm(2)
                total_norm += param_norm.item() ** 2
                param_count += 1

                # Store individual layer statistics
                gradient_stats[f"grad_norm_{name}"] = param_norm.item()

                # Store gradient statistics
                grad_data = param.grad.data.cpu().numpy()
                gradient_stats[f"grad_mean_{name}"] = float(np.mean(grad_data))
                gradient_stats[f"grad_std_{name}"] = float(np.std(grad_data))

        total_norm = total_norm ** (0.5)
        gradient_stats["gradient_norm"] = total_norm
        gradient_stats["gradient_count"] = param_count

        return gradient_stats

        max_grads = []
        layers = []

                layers.append(name)
```

### Advanced Data Augmentation Strategies

The augmentation system implements adaptive pipelines that adjust transformation intensity based on training progress and model performance. It supports geometric, color, noise, and advanced transformations optimized for computer vision tasks.

**Adaptive Augmentation Pipeline**

```python
import albumentations as A
from albumentations.pytorch import ToTensorV2
import cv2
import numpy as np
import torch
from typing import Dict, List, Optional, Union, Tuple
import random
from PIL import Image

class AdaptiveAugmentationPipeline:
    """
    Intelligent augmentation system that adapts strategies based on 
    training progress, data characteristics, and model performance
    """

    def __init__(self, 
                 input_size: Tuple[int, int] = (224, 224),
                 task_type: str = "classification",
                 difficulty_level: str = "medium"):

        self.input_size = input_size
        self.task_type = task_type
        self.difficulty_level = difficulty_level

        # Augmentation strength progression
        self.augmentation_schedules = {
            "conservative": [0.2, 0.3, 0.4, 0.3, 0.2],  # Mild augmentation
            "progressive": [0.1, 0.3, 0.6, 0.8, 0.5],   # Increase then decrease
            "aggressive": [0.6, 0.7, 0.8, 0.8, 0.6],    # Strong throughout
            "adaptive": []  # Will be computed based on performance
        }

        # Performance tracking for adaptation
        self.performance_history = []
        self.current_strategy = "progressive"

        # Create augmentation pools
        self.geometric_transforms = self._create_geometric_pool()
        self.color_transforms = self._create_color_pool()
        self.noise_transforms = self._create_noise_pool()
        self.advanced_transforms = self._create_advanced_pool()

    def get_augmentation_pipeline(self, 
                                epoch: int, 
                                total_epochs: int,
                                current_performance: Optional[float] = None) -> A.Compose:
        """
        Generate augmentation pipeline adapted to current training stage
        """

        # Update performance tracking
        if current_performance is not None:
            self.performance_history.append(current_performance)
            self._adapt_strategy()

        # Calculate augmentation strength
        strength = self._calculate_augmentation_strength(epoch, total_epochs)

        # Select transforms based on strategy and strength
        transforms = self._select_transforms(strength)

        # Create pipeline
        pipeline = A.Compose(
            transforms,
            additional_targets={},
            p=1.0
        )

        return pipeline

    def _calculate_augmentation_strength(self, epoch: int, total_epochs: int) -> float:
        """Calculate current augmentation strength based on training progress"""

        progress = epoch / total_epochs
        schedule = self.augmentation_schedules[self.current_strategy]

        if self.current_strategy == "adaptive":
            # Adaptive strategy based on performance
            strength = self._adaptive_strength_calculation(progress)
        else:
            # Use predefined schedule
            phase_idx = min(int(progress * len(schedule)), len(schedule) - 1)
            strength = schedule[phase_idx]

        return strength

    def _adaptive_strength_calculation(self, progress: float) -> float:
        """Calculate adaptive augmentation strength based on model performance"""

        if len(self.performance_history) < 5:
            return 0.3  # Default moderate strength

        recent_performance = self.performance_history[-5:]
        performance_trend = np.mean(np.diff(recent_performance))
        current_performance = recent_performance[-1]

        # Adaptive logic
        if performance_trend > 0.01:  # Good improvement
            # Gradually increase augmentation
            return min(0.8, 0.3 + progress * 0.4)
        elif performance_trend < -0.01:  # Performance declining
            # Reduce augmentation strength
            return max(0.1, 0.3 - progress * 0.2)
        else:  # Stable performance
            # Maintain moderate augmentation
            return 0.4 + 0.2 * math.sin(progress * math.pi)

    def _select_transforms(self, strength: float) -> List:
        """Select appropriate transforms based on augmentation strength"""

        transforms = []

        # Basic preprocessing (always included)
        transforms.extend([
            A.Resize(self.input_size[0], self.input_size[1]),
        ])

        # Geometric transforms (scaled by strength)
        if strength > 0.1:
            geometric_selection = self._weighted_selection(
                self.geometric_transforms, 
                int(3 * strength)
            )
            transforms.extend(geometric_selection)

        # Color transforms
        if strength > 0.2:
            color_selection = self._weighted_selection(
                self.color_transforms,
                int(2 * strength)
            )
            transforms.extend(color_selection)

        # Noise and advanced transforms for higher strength
        if strength > 0.5:
            noise_selection = self._weighted_selection(
                self.noise_transforms,
                max(1, int((strength - 0.5) * 4))
            )
            transforms.extend(noise_selection)

        if strength > 0.7:
            advanced_selection = self._weighted_selection(
                self.advanced_transforms,
                max(1, int((strength - 0.7) * 3))
            )
            transforms.extend(advanced_selection)

        # Final normalization and tensor conversion
        transforms.extend([
            A.Normalize(
                mean=[0.485, 0.456, 0.406],
                std=[0.229, 0.224, 0.225]
            ),
            ToTensorV2()
        ])

        return transforms

    def _create_geometric_pool(self) -> List[Tuple[any, float]]:

    def _weighted_selection(self, 
                          transform_pool: List[Tuple[any, float]], 
                          num_select: int) -> List:
        """Select transforms based on weights"""

        if num_select >= len(transform_pool):
            return [t[0] for t in transform_pool]

        # Extract weights
        weights = [t[1] for t in transform_pool]

        # Weighted random selection without replacement
        selected_indices = np.random.choice(
            len(transform_pool), 
            size=min(num_select, len(transform_pool)), 
            replace=False, 
            p=np.array(weights) / np.sum(weights)
        )

        return [transform_pool[i][0] for i in selected_indices]

    def _adapt_strategy(self):
        """Adapt augmentation strategy based on performance trends"""
        if len(self.performance_history) >= 10:
            recent_trend = np.mean(np.diff(self.performance_history[-5:]))
            if recent_trend < -0.01:
                self.current_strategy = "conservative"
            elif recent_trend > 0.01:
                self.current_strategy = "aggressive"
            else:
                self.current_strategy = "progressive"

    def create_mixup_augmentation(self, alpha: float = 0.2):
        """Create MixUp augmentation for training"""

        def mixup_data(x, y, alpha=alpha):
            """Apply MixUp augmentation to batch"""
            if alpha > 0:
                lam = np.random.beta(alpha, alpha)
            else:
                lam = 1

            batch_size = x.size(0)
            index = torch.randperm(batch_size).to(x.device)

            mixed_x = lam * x + (1 - lam) * x[index, :]
            y_a, y_b = y, y[index]

            return mixed_x, y_a, y_b, lam

        return mixup_data

    def create_cutmix_augmentation(self, alpha: float = 1.0):
        """Create CutMix augmentation for training"""

        def cutmix_data(x, y, alpha=alpha):
            """Apply CutMix augmentation to batch"""
            if alpha > 0:
                lam = np.random.beta(alpha, alpha)
            else:
                lam = 1

            batch_size = x.size(0)
            index = torch.randperm(batch_size).to(x.device)

            # Generate random bbox
            _, _, H, W = x.shape
            cut_rat = np.sqrt(1.0 - lam)
            cut_w = int(W * cut_rat)
            cut_h = int(H * cut_rat)

            # Uniform
            cx = np.random.randint(W)
            cy = np.random.randint(H)

            bbx1 = np.clip(cx - cut_w // 2, 0, W)
            bby1 = np.clip(cy - cut_h // 2, 0, H)
            bbx2 = np.clip(cx + cut_w // 2, 0, W)
            bby2 = np.clip(cy + cut_h // 2, 0, H)

            x[:, :, bby1:bby2, bbx1:bbx2] = x[index, :, bby1:bby2, bbx1:bbx2]

            # Adjust lambda to exactly match pixel ratio
            lam = 1 - ((bbx2 - bbx1) * (bby2 - bby1) / (W * H))

            y_a, y_b = y, y[index]

            return x, y_a, y_b, lam

        return cutmix_data
```

                result = self._execute_optimization_step(step)
                results[step] = result
                self.logger.info(f"Completed {step}: {result.get('summary', 'Success')}")
            except Exception as e:
                self.logger.error(f"Failed {step}: {str(e)}")
                results[step] = {"success": False, "error": str(e)}

        # Generate optimization report
        optimization_report = self._generate_optimization_report(results)

        return optimization_report

        """Execute individual optimization step"""

        if step == "torch_jit_optimization":
            return self._torch_jit_optimization()
        elif step == "onnx_export_optimization":
            return self._onnx_export_optimization()

    def _benchmark_model(self, 
                        model: torch.nn.Module, 
                        dummy_input: torch.Tensor, 
                        num_runs: int = 100) -> float:
        """Benchmark model inference time"""

        model.eval()

        # Warmup
        for _ in range(10):
            with torch.no_grad():
                _ = model(dummy_input)

        # Benchmark
        start_time = time.time()
        with torch.no_grad():
            for _ in range(num_runs):
                _ = model(dummy_input)
        end_time = time.time()

        return (end_time - start_time) / num_runs

        # Warmup

        end_time = time.time()

        return (end_time - start_time) / num_runs

    def _get_model_size(self, model: torch.nn.Module) -> float:
        """Calculate model size in MB"""

        param_size = 0
        for param in model.parameters():
            param_size += param.nelement() * param.element_size()

        buffer_size = 0
        for buffer in model.buffers():
            buffer_size += buffer.nelement() * buffer.element_size()

        size_mb = (param_size + buffer_size) / (1024 * 1024)  # Convert to MB
        return round(size_mb, 2)

**Challenge 3: Experiment Reproducibility**

**Solution**: Comprehensive environment capture and seeding:

```python
import torch
import random
import numpy as np
import os
import subprocess
import sys

def capture_environment() -> Dict:
    """Capture complete training environment for reproducibility"""
    return {
        'pytorch_version': torch.__version__,
        'cuda_version': torch.version.cuda if torch.cuda.is_available() else None,
        'python_version': sys.version,
        'git_commit': _get_git_commit_hash(),
        'random_seed': torch.initial_seed(),
        'platform': sys.platform,
        'dependencies': _get_dependency_versions()
    }

def set_reproducible_seed(seed: int = 42):
    """Set reproducible random seeds across all libraries"""
    random.seed(seed)
    np.random.seed(seed)
    torch.manual_seed(seed)
    torch.cuda.manual_seed_all(seed)
    os.environ['PYTHONHASHSEED'] = str(seed)
    
    # Ensure deterministic behavior
    torch.backends.cudnn.deterministic = True
    torch.backends.cudnn.benchmark = False

def _get_git_commit_hash() -> str:
    """Get current git commit hash"""
    try:
        return subprocess.check_output(['git', 'rev-parse', 'HEAD']).decode().strip()
    except:
        return "unknown"

def _get_dependency_versions() -> Dict:
    """Get versions of key dependencies"""
    versions = {}
    packages = ['torch', 'optuna', 'mlflow', 'albumentations']
    
    for pkg in packages:
        try:
            versions[pkg] = __import__(pkg).__version__
        except:
            versions[pkg] = "not installed"
    
    return versions
```
