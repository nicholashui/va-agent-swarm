import cv2
import numpy as np
from scipy.signal import find_peaks

def analyze_motion_intensity(video_path):
    cap = cv2.VideoCapture(video_path)
    motion_scores = []
    prev_frame = None
    
    while cap.isOpened():
        ret, frame = cap.read()
        if not ret:
            break
            
        gray = cv2.cvtColor(frame, cv2.COLOR_BGR2GRAY)
        gray = cv2.GaussianBlur(gray, (5, 5), 0)
        
        if prev_frame is not None:
            # Calculate optical flow for motion vectors
            flow = cv2.calcOpticalFlowFarneback(
                prev_frame, gray, None, 0.5, 3, 15, 3, 5, 1.2, 0
            )
            
            # Motion magnitude and edge density
            magnitude, _ = cv2.cartToPolar(flow[..., 0], flow[..., 1])
            avg_motion = np.mean(magnitude)
            edges = cv2.Canny(gray, 50, 150)
            edge_density = np.sum(edges > 0) / (edges.shape[0] * edges.shape[1])
            
            action_score = avg_motion * edge_density
            motion_scores.append(action_score)
        
        prev_frame = gray
    
    # Identify action peaks
    peaks, _ = find_peaks(motion_scores, 
                         height=np.mean(motion_scores) + np.std(motion_scores))
    
    return {
        "motion_timeline": motion_scores,
        "action_peaks": peaks,
        "peak_timestamps": [i / 24 for i in peaks]  # Convert to seconds at 24 fps
    }

# Key action moments identified:
# 10:25 - Car chase begins (high motion peak)
# 31:15 - Warehouse shootout
# 83:20 - Final confrontation climax
```

import librosa
import numpy as np

def analyze_audio_drama(audio_path):
    y, sr = librosa.load(audio_path, sr=22050)
    
    # Calculate RMS energy and spectral features
    rms = librosa.feature.rms(y=y, frame_length=2048, hop_length=512)[0]
    rms_db = 20 * np.log10(rms + 1e-6)
    
    spectral_centroid = librosa.feature.spectral_centroid(y=y, sr=sr)[0]
    onset_envelope = librosa.onset.onset_strength(y=y, sr=sr)
    
    times = librosa.frames_to_time(np.arange(len(rms)), sr=sr, hop_length=512)
    
    return {
        "times": times,
        "loudness_db": rms_db,
        "spectral_brightness": spectral_centroid,
        "onset_strength": onset_envelope,
        "drama_timeline": rms_db * (spectral_centroid / 1000)  # Combined drama score
    }

# Analyze separate stems

def analyze_trailer_dialogue(transcription):
    """Extract dramatic dialogue suitable for trailers"""
    
    dramatic_keywords = {
        "conflict": ["betrayed", "war", "fight", "enemy", "destroy", "kill"],
        "emotion": ["love", "heart", "tears", "afraid", "hope", "family"],
        "mystery": ["truth", "secret", "hidden", "conspiracy", "lies"],
        "urgency": ["now", "time", "hurry", "last chance", "never", "forever"],
        "threat": ["danger", "death", "end", "doom", "terror", "nightmare"]
    }
    
    dialogue_analysis = []
    
    for segment in transcription["segments"]:
        text = segment["text"].strip()
        start_time = segment["start"]
        end_time = segment["end"]
        
        # Calculate emotional weight
        emotion_score = 0
        category_scores = {}
        
        for category, keywords in dramatic_keywords.items():
            category_score = sum(1 for keyword in keywords 
                               if keyword.lower() in text.lower())
            category_scores[category] = category_score
            emotion_score += category_score
        
        # Check dramatic patterns
        dramatic_patterns = [
            r"I (?:will|must|have to|need to)",  # Determination

            r"(?:You|We) (?:can't|won't|must)",  # Commands/refusals
            r"It's (?:over|too late|impossible)", # Finality
            r"(?:Save|Protect|Destroy) (?:the|her|him|them)", # Stakes
        ]
        
        pattern_matches = sum(1 for pattern in dramatic_patterns 
                            if re.search(pattern, text, re.IGNORECASE))
        
        # Trailer suitability score
        word_count = len(text.split())
        trailer_score = (
            emotion_score * 2 +           # Emotional content
            pattern_matches * 3 +         # Dramatic structure
            (8 - abs(word_count - 8)) * 1  # Optimal length preference
        )
        
        dialogue_analysis.append({
            "text": text,
            "start_time": start_time,
            "end_time": end_time,
            "duration": end_time - start_time,
            "emotion_score": emotion_score,
            "pattern_matches": pattern_matches,
            "trailer_score": trailer_score,
            "word_count": word_count
        })
    
    # Select top dramatic lines
    best_dialogue = sorted(dialogue_analysis, 
                          key=lambda x: x["trailer_score"], 
                          reverse=True)[:15]
    
    return best_dialogue

import cv2
from sklearn.cluster import KMeans

def analyze_visual_impact(video_path):
    cap = cv2.VideoCapture(video_path)
    visual_data = []
    
    while cap.isOpened():
        ret, frame = cap.read()
        if not ret:
            break
        
        # HSV color analysis
        hsv = cv2.cvtColor(frame, cv2.COLOR_BGR2HSV)
        saturation = hsv[:, :, 1]
        avg_saturation = np.mean(saturation)
        
        # Color diversity
        pixels = frame.reshape((-1, 3))
        kmeans = KMeans(n_clusters=8, random_state=42, n_init=10)
        kmeans.fit(pixels)
        colors = kmeans.cluster_centers_
        
        # Calculate color harmony
        color_distances = []
        for i in range(len(colors)):
            for j in range(i+1, len(colors)):
                dist = np.linalg.norm(colors[i] - colors[j])
                color_distances.append(dist)
        
        color_harmony = np.mean(color_distances)
        
        # Brightness and contrast
        gray = cv2.cvtColor(frame, cv2.COLOR_BGR2GRAY)
        brightness = np.mean(gray)
        contrast = gray.std()
        
        visual_data.append({
            "avg_saturation": avg_saturation,
            "color_harmony": color_harmony,
            "brightness": brightness,
            "contrast": contrast,
            "visual_impact": avg_saturation * contrast / 100
        })
    
    return visual_data
```

```python
def rank_scenes_for_trailer(scene_metadata, motion_data, audio_data, 
                           visual_data, dialogue_data):
    """Comprehensive scene ranking using weighted scoring"""
    
    ranked_scenes = []
    
    for scene_id in scene_metadata.keys():
        scene_start = scene_id * 8  # Average scene duration
        scene_end = (scene_id + 1) * 8
        
        # Extract relevant data for this scene
        scene_motion = np.mean([
            d["action_score"] for d in motion_data 
            if scene_start <= d["timestamp"] <= scene_end
        ])
        
        scene_audio = np.mean([
            d["drama_score"] for d in audio_data
            if scene_start <= d["times"] <= scene_end
        ]) if audio_data else 0
        
        scene_visual = np.mean([
            d["visual_impact"] for d in visual_data
            if scene_start <= i <= scene_end for i, d in enumerate(visual_data)
        ]) if visual_data else 0
        
        # Best dialogue in this scene
        scene_dialogue = max([
            d["trailer_score"] for d in dialogue_data
            if scene_start <= d["start_time"] <= scene_end
        ] + [0])
        
        # CLIP semantic understanding
        clip_score = scene_metadata[scene_id]["confidence"]
        clip_category = scene_metadata[scene_id]["category"]
        
        # Weighted final score
        final_score = (
            scene_motion * 0.40 +      # 40% motion/action
            scene_audio * 0.30 +       # 30% audio drama
            scene_visual * 0.20 +      # 20% visual impact
            scene_dialogue * 0.10 +    # 10% dialogue quality
            clip_score * 0.15          # 15% semantic relevance (bonus)
        )
        
        ranked_scenes.append({
            "scene_id": scene_id,
            "timestamp": scene_start,
            "final_score": final_score,
            "motion_score": scene_motion,
            "audio_score": scene_audio,
            "visual_score": scene_visual,
            "dialogue_score": scene_dialogue,
            "clip_score": clip_score,
            "category": clip_category,
            "recommended_duration": min(12, max(3, final_score * 2))  # 3-12 seconds
        })
    
    # Sort by final score
    ranked_scenes.sort(key=lambda x: x["final_score"], reverse=True)
    
    # Ensure scene variety (max 3 per category)
    diverse_scenes = ensure_scene_diversity(ranked_scenes)
    
    return diverse_scenes[:15]  # Top 15 scenes for 2.5-minute trailer

def ensure_scene_diversity(scenes, max_per_category=3):
    """Maintain variety across scene types"""
    category_counts = {}
    selected_scenes = []
    
    for scene in scenes:
        category = scene["category"]
        if category_counts.get(category, 0) < max_per_category:
            selected_scenes.append(scene)
            category_counts[category] = category_counts.get(category, 0) + 1
    
    return selected_scenes
```

import librosa
import numpy as np
from scipy import signal

def analyze_trailer_music(music_path):
    """Comprehensive music analysis for editing synchronization"""
    
    # Load high-quality audio
    y, sr = librosa.load(music_path, sr=44100)
    
    # Basic beat tracking
    tempo, beat_frames = librosa.beat.beat_track(y=y, sr=sr, start_bpm=120)
    beat_times = librosa.frames_to_time(beat_frames, sr=sr)
    
    # Advanced onset detection
    hop_length = 512
    onset_envelope = librosa.onset.onset_strength(
        y=y, sr=sr, aggregate=np.median, fmax=8000, hop_length=hop_length
    )
    onsets = librosa.onset.onset_detect(
        onset_envelope=onset_envelope, sr=sr, units='time'
    )
    
    # Tempo variations and structure
    oenv = librosa.onset.onset_strength(y=y, sr=sr, hop_length=hop_length)
    tempogram = librosa.feature.tempogram(onset_envelope=oenv, sr=sr, hop_length=hop_length)
    
    # Chroma features for harmonic analysis
    chroma = librosa.feature.chroma_cqt(y=y, sr=sr, hop_length=hop_length)
    
    # Structural segmentation
    features = np.vstack([chroma, tempogram])
    from sklearn.cluster import KMeans
    features_t = features.T
    n_clusters = 4  # Intro, build, climax, outro
    
    kmeans = KMeans(n_clusters=n_clusters, random_state=42)
    segments = kmeans.fit_predict(features_t)
    
    times = librosa.frames_to_time(np.arange(len(segments)), sr=sr, hop_length=hop_length)
    section_changes = np.where(np.diff(segments) != 0)[0]
    section_times = times[section_changes]
    
    # Music structure for epic trailer (typical 120 BPM orchestral build)
    music_structure = {
        "intro": {"start": 0, "end": 15, "style": "soft_piano", "bpm": 60},
        "build": {"start": 15, "end": 60, "style": "rising_strings", "bpm": 120},
        "climax": {"start": 60, "end": 90, "style": "full_orchestra", "bpm": 120},
        "outro": {"start": 90, "end": 120, "style": "resolution", "bpm": 80}
    }
    
    return {
        "tempo": tempo,
        "beat_times": beat_times,
        "onset_times": onsets,
        "segments": segments.tolist(),
        "section_times": section_times.tolist(),
        "music_structure": music_structure,
        "energy_profile": librosa.feature.rms(y=y, hop_length=hop_length)[0],
        "onset_envelope": onset_envelope
    }

# Typical trailer music analysis results:
# Detected tempo: 120.3 BPM
# Total beats: 240 (4 beats per measure × 60 measures)
# Section boundaries: [15.2s, 60.1s, 90.4s] (build, climax, outro)

from moviepy.editor import VideoFileClip, concatenate_videoclips

def synchronize_scenes_to_music(selected_scenes, music_analysis, target_duration=150):
    """Create beat-synchronized editing timeline"""
    
    beat_times = music_analysis["beat_times"]
    section_times = music_analysis["section_times"]
    music_structure = music_analysis["music_structure"]
    
    # Define editing styles per music section
    editing_styles = {
        "intro": {
            "style": "slow_build",
            "cut_frequency": 4,  # One cut every 4 beats (slow)
            "scene_types": ["establishing", "emotional"],
            "cut_duration": 8.0  # Longer shots
        },
        "build": {
            "style": "accelerating", 
            "cut_frequency": 2,  # One cut every 2 beats (medium)
            "scene_types": ["dramatic", "suspense"],
            "cut_duration": 4.0  # Medium shots
        },
        "climax": {
            "style": "rapid_fire",
            "cut_frequency": 0.5,  # Two cuts per beat (fast)
            "scene_types": ["action", "confrontation"],
            "cut_duration": 1.0  # Quick cuts
        },
        "outro": {
            "style": "resolution",
            "cut_frequency": 3,  # One cut every 3 beats (medium-slow)
            "scene_types": ["emotional", "final_reveal"],
            "cut_duration": 6.0  # Resolution shots
        }
    }
    
    # Sort scenes by dramatic intensity for optimal placement
    scenes_by_intensity = sorted(selected_scenes, 
                               key=lambda x: x["final_score"], reverse=True)
    
    timeline = []
    scene_index = 0
    current_time = 0
    
    # Process each music section
    for section_name, section_data in music_structure.items():
        if scene_index >= len(scenes_by_intensity):
            break
            
        section_start = section_data["start"]
        section_end = section_data["end"]
        style = editing_styles.get(section_name, editing_styles["build"])
        
        # Select appropriate scenes for this section
        section_scenes = [
            s for s in scenes_by_intensity[scene_index:]
            if any(st in s["category"].lower() for st in style["scene_types"])
        ]
        
        # Calculate cut points
        beats_in_section = [
            bt for bt in beat_times 
            if section_start <= bt <= section_end
        ]
        
        cut_interval = 60 / music_analysis["tempo"] * style["cut_frequency"]
        
        for beat_time in beats_in_section[::int(style["cut_frequency"])]:
            if current_time >= target_duration or scene_index >= len(scenes_by_intensity):
                break
                
            if beat_time > current_time:
                # Insert scene at this beat
                scene = scenes_by_intensity[scene_index]
                scene_duration = min(
                    style["cut_duration"], 
                    section_end - beat_time
                )
                
                timeline.append({
                    "scene_id": scene["scene_id"],
                    "source_timestamp": scene["timestamp"],
                    "start_time": beat_time,
                    "duration": scene_duration,
                    "category": scene["category"],
                    "section": section_name,
                    "beat_aligned": True,
                    "intensity": scene["final_score"],
                    "style": style["style"]
                })
                
                current_time = beat_time + scene_duration
                scene_index += 1
    
    # Ensure timeline fills target duration
    if current_time < target_duration:
        # Extend final scene or add outro
        if timeline:
            timeline[-1]["duration"] += target_duration - current_time
    
    return timeline

# Generate synchronized timeline

def assemble_synchronized_trailer(timeline, source_video_path, music_path, output_path):
    """Assemble final trailer with precise beat timing"""
    
    source_video = VideoFileClip(source_video_path, audio=False)  # Video only
    clips = []
    
    for cut_info in timeline:
        scene_id = cut_info["scene_id"]
        start_time = cut_info["start_time"]
        duration = cut_info["duration"]
        source_timestamp = cut_info["source_timestamp"]
        
        # Calculate source clip timing with padding for smooth editing
        clip_start = max(0, source_timestamp - 1)  # 1s before for transitions
        clip_end = min(source_video.duration, source_timestamp + duration + 1)
        
        # Extract subclip
        scene_clip = source_video.subclip(clip_start, clip_end)
        
        # Trim to exact duration needed
        if scene_clip.duration > duration:
            scene_clip = scene_clip.subclip(0, duration)
        else:
            # Handle short clips with speed adjustment or looping
            if duration / scene_clip.duration < 2.0:  # Don't slow too much
                speed_factor = scene_clip.duration / duration
                scene_clip = scene_clip.fx(lambda c: c.speedx(speed_factor))
            else:
                # Loop if still too short
                scene_clip = scene_clip.loop(duration=duration)
        
        # Apply section-specific effects
        if "climax" in cut_info["section"]:
            # Faster motion blur for action
            scene_clip = scene_clip.fx(lambda c: c.shake())
        elif "intro" in cut_info["section"]:
            # Subtle zoom for establishing shots
            scene_clip = scene_clip.fx(lambda c: c.zoom_in(0.1))
        
        clips.append(scene_clip)
    
    # Concatenate all clips
    trailer_video = concatenate_videoclips(clips, method="compose")
    
    # Add synchronized music
    music_clip = AudioFileClip(music_path).subclip(0, trailer_video.duration)
    final_trailer = trailer_video.set_audio(music_clip)
    
    # Export with professional settings
    final_trailer.write_videofile(

        fps=24,
        codec='libx264',
        audio_codec='aac',
        bitrate="5000k",
        temp_audiofile='temp-audio.m4a',
        remove_temp=True,
        verbose=False,
        logger=None
    )
    
    # Clean up
    source_video.close()
    music_clip.close()
    final_trailer.close()
    
    return output_path

def synchronize_titles_to_timeline(title_cards, music_analysis, video_timeline):
    """Place title cards at optimal musical moments"""
    
    beat_times = music_analysis["beat_times"]
    music_structure = music_analysis["music_structure"]
    
    # Define optimal placement windows
    title_placement = {
        "main_tagline": {
            "preferred_section": "build",
            "min_duration": 4,
            "max_duration": 6,
            "beat_offset": 2  # After strong beat
        },
        "director_credit": {
            "preferred_section": "intro", 
            "min_duration": 3,
            "max_duration": 4,
            "beat_offset": 1
        },
        "cast_reveal": {
            "preferred_section": "build",
            "min_duration": 3,
            "max_duration": 4,
            "beat_offset": 0
        },
        "release_info": {
            "preferred_section": "outro",
            "min_duration": 5,
            "max_duration": 7,
            "beat_offset": 4  # Major resolution beat
        }
    }
    
    synchronized_titles = []
    
    for title_name, title_data in title_cards.items():
        placement = title_placement.get(title_name, title_placement["director_credit"])
        
        # Find appropriate placement window
        section_start = music_structure[placement["preferred_section"]]["start"]
        section_end = music_structure[placement["preferred_section"]]["end"]
        
        # Find suitable gap in video timeline
        available_slots = []
        for i in range(len(video_timeline) - 1):
            current_end = video_timeline[i]["start_time"] + video_timeline[i]["duration"]
            next_start = video_timeline[i + 1]["start_time"]
            
            gap_duration = next_start - current_end
            gap_start = current_end
            
            # Check if gap fits title requirements
            if (section_start <= gap_start <= section_end and 
                gap_duration >= placement["min_duration"]):
                
                # Find nearest beat for synchronization
                section_beats = [bt for bt in beat_times 
                               if section_start <= bt <= section_end]
                
                if section_beats:
                    nearest_beat = min(section_beats, 
                                     key=lambda x: abs(x - (gap_start + placement["beat_offset"] * (60/music_analysis["tempo"]))))
                    
                    available_slots.append({
                        "start_time": nearest_beat,
                        "duration": min(gap_duration, placement["max_duration"]),
                        "gap_score": abs(nearest_beat - gap_start),
                        "section": placement["preferred_section"]
                    })
        
        # Select best placement
        if available_slots:
            best_slot = min(available_slots, key=lambda x: x["gap_score"])
            
            synchronized_titles.append({
                "title_name": title_name,
                "start_time": best_slot["start_time"],
                "duration": best_slot["duration"],
                "placement_data": title_data,
                "music_section": best_slot["section"],
                "beat_sync": True
            })
    
    return synchronized_titles

def process_music_for_trailer(music, timeline, sr):
    """Apply trailer-specific music processing"""
    
    # Multiband compression for cinematic sound
    low_freq, mid_freq, high_freq = 250, 4000, 8000
    
    # Create crossover filters
    sos_low = signal.butter(6, low_freq, btype='low', fs=sr, output='sos')
    sos_mid = signal.butter(6, [low_freq, mid_freq], btype='band', fs=sr, output='sos')
    sos_high = signal.butter(6, high_freq, btype='high', fs=sr, output='sos')
    
    # Split into frequency bands
    low_band = signal.sosfilt(sos_low, music, axis=1)
    mid_band = signal.sosfilt(sos_mid, music, axis=1)
    high_band = signal.sosfilt(sos_high, music, axis=1)
    
    # Apply genre-specific compression

    mid_compressed = apply_multiband_compression(mid_band, threshold=-20, ratio=6, sr=sr)

def apply_multiband_compression(audio, threshold, ratio, sr):
    """Apply frequency-specific compression"""
    
    # Convert to dB
    audio_db = 20 * np.log10(np.abs(audio) + 1e-10)
    
    # Gain reduction calculation
    over_threshold = audio_db > threshold
    gain_reduction = np.zeros_like(audio_db)
    
    gain_reduction[over_threshold] = (
        (audio_db[over_threshold] - threshold) * (1 - 1/ratio)
    )
    
    # Apply attack/release smoothing (10ms attack, 100ms release)
    attack_samples = int(0.01 * sr)
    release_samples = int(0.1 * sr)
    
    for channel in range(audio.shape[0]):
        gain_reduction[channel] = smooth_gain_reduction(
            gain_reduction[channel], attack_samples, release_samples
        )
    
    # Convert back to linear gain
    gain_linear = 10 ** (gain_reduction / 20)
    compressed = audio * gain_linear
    
    return compressed

def intelligent_audio_ducking(music, dialogue, sfx, sr):
    """Smart ducking based on content priority"""
    
    # Analyze dialogue presence (primary element)
    dialogue_rms = librosa.feature.rms(y=dialogue, hop_length=512)[0]
    dialogue_db = 20 * np.log10(dialogue_rms + 1e-10)
    dialogue_threshold = np.percentile(dialogue_db, 75)  # Active speech threshold
    
    # Create time-smoothed dialogue envelope
    dialogue_envelope = np.convolve(
        (dialogue_db > dialogue_threshold).astype(float),
        np.ones(1024)/1024,  # 21ms smoothing window at 48kHz
        mode='same'
    )
    
    # Calculate ducking curves (6dB music reduction, 3dB SFX reduction during speech)
    music_ducking = 1.0 - (dialogue_envelope * 0.5)  # Max 6dB reduction
    sfx_ducking = 1.0 - (dialogue_envelope * 0.25)    # Max 3dB reduction
    
    # Frequency-specific ducking (protect dialogue range 200-4000Hz)
    music_ducked = frequency_aware_ducking(music, music_ducking, dialogue_envelope, sr)
    sfx_ducked = sfx * sfx_ducking
    
    # Mix levels (dialogue dominant, music support, sfx accent)
    dialogue_level = 0.0  # Dialogue at 0 dBFS reference
    music_level = -12     # Music at -12 dBFS
    sfx_level = -18       # SFX at -18 dBFS
    
    # Apply final mixing
    final_mix = (
        dialogue * dialogue_level +
        music_ducked * music_level +
        sfx_ducked * sfx_level
    )
    
    # Stereo enhancement
    final_mix = stereo_imaging(final_mix)
    
    return final_mix

def master_for_broadcast(audio, sr, target_lufs=-14, max_peak=-1):
    """Apply broadcast-standard loudness normalization"""
    
    # Measure integrated LUFS
    current_lufs = measure_lufs(audio, sr)
    
    # Calculate required gain
    gain_db = target_lufs - current_lufs
    gain_linear = 10 ** (gain_db / 20)
    
    # Apply gain
    normalized = audio * gain_linear
    
    # True peak limiting
    peak_limited = true_peak_limiter(normalized, max_peak, sr)
    
    # Final safety clip
    final_mix = np.clip(peak_limited, -1.0, 1.0)
    
    # Verify final loudness
    final_lufs = measure_lufs(final_mix, sr)
    final_peak = 20 * np.log10(np.max(np.abs(final_mix)))
    
    print(f"Mastered: {final_lufs:.1f} LUFS, {final_peak:.1f} dBTP")
    
    return final_mix

def measure_lufs(audio, sr, block_size=400):  # 400ms blocks per EBU R128
    """Calculate Loudness Units relative to Full Scale (ITU-R BS.1770)"""
    
    # K-weighting filter (high shelf + high pass)
    # High shelf filter (4kHz)
    b_hf, a_hf = signal.iirfilter(
        2, 7380, btype='high', analog=True, ftype='butter'
    )
    b_hf, a_hf = signal.bilinear(b_hf, a_hf, fs=sr)
    
    # High pass filter (50Hz)
    b_hp, a_hp = signal.iirfilter(
        2, 50, btype='high', analog=True, ftype='butter'
    )
    b_hp, a_hp = signal.bilinear(b_hp, a_hp, fs=sr)
    
    # Apply weighting filters
    weighted_low = signal.filtfilt(b_hp, a_hp, audio, axis=1)
    weighted = signal.filtfilt(b_hf, a_hf, weighted_low, axis=1)
    
    # Calculate mean square
    mean_square = np.mean(weighted**2, axis=0)
    
    # Gated integration (absolute threshold -70 LUFS)
    block_samples = int(block_size * sr / 1000)
    blocks = []
    
    for i in range(0, len(mean_square) - block_samples, block_samples // 4):
        block_power = np.mean(mean_square[i:i+block_samples])
        if block_power > 0:
            block_lufs = -0.691 + 10 * np.log10(block_power)  # 0.691 = 10*log10(2**23)
            
            if block_lufs > -70:  # Absolute gating
                blocks.append(block_lufs)
    
    if not blocks:
        return -70.0
    
    # Relative gating (-10 LUFS below ungated mean)
    ungated_mean = np.mean(blocks)
    relative_threshold = ungated_mean - 10
    gated_blocks = [b for b in blocks if b > relative_threshold]
    
    if gated_blocks:
        integrated_lufs = np.mean(gated_blocks)
    else:
        integrated_lufs = ungated_mean
    
    return integrated_lufs

def apply_platform_optimizations(video_clip, specs):
    """Apply platform-specific video processing"""
    
    # Duration adaptation
    if video_clip.duration > specs["duration"]:
        # Smart trimming - preserve key moments
        video_clip = smart_trim_to_duration(video_clip, specs["duration"], specs["editing_style"])
    elif video_clip.duration < specs["duration"]:
        # Loop or extend final moment
        video_clip = extend_to_duration(video_clip, specs["duration"])
    
    # Resolution and aspect ratio
    target_width, target_height = specs["resolution"]
    
    if specs["aspect_ratio"] < 1.0:  # Vertical format
        # Smart cropping for vertical platforms
        resized = video_clip.resize(height=target_height)
        w, h = resized.size
        
        # AI-guided focal point cropping using CLIP
        crop_x = calculate_vertical_crop_center(resized, specs)
        cropped = resized.crop(x1=crop_x - target_width//2, y1=0, 
                              x2=crop_x + target_width//2, y2=target_height)
        
        optimized = cropped.resize((target_width, target_height))
        
        # Add social media overlays if specified
        if "social_overlays" in specs:
            optimized = add_platform_branding(optimized, specs["platform_branding"])
            
    else:  # Horizontal format
        # Standard letterboxing or pillarboxing
        resized = video_clip.resize(width=target_width)
        
        if resized.h < target_height:
            # Letterbox (black bars top/bottom)
            optimized = resized.on_color(
                size=(target_width, target_height), 
                color=(0, 0, 0), col_opacity=1
            )
        else:
            # Pillarbox (black bars sides)
            w, h = resized.size
            crop_y = h//2 - target_height//2
            optimized = resized.crop(y1=crop_y, y2=crop_y + target_height)
            optimized = optimized.resize((target_width, target_height))
    
    # Frame rate conversion
    if abs(optimized.fps - specs["framerate"]) > 0.1:
        optimized = optimized.set_fps(specs["framerate"])
    
    # Color space conversion
    if specs["color_space"] == "BT.709":  # Standard for broadcast/web
        # Apply Rec.709 color matrix if source is film/DCI-P3
        pass  # Implementation depends on source color space
    elif specs["color_space"] == "BT.2020":  # Theatrical 4K
        # Preserve wide color gamut
        pass
    
    # Platform-specific effects
    if specs["editing_style"] == "rapid_highlights":
        # TV spot: Faster pacing, higher contrast
        optimized = optimized.fx(lambda c: c.adjust_contrast(1.2))
        optimized = optimized.fx(lambda c: c.speedx(1.2))  # 20% faster
        
    elif specs["editing_style"] == "attention_grab":
        # Social: Immediate impact, text overlays
        optimized = add_attention_hooks(optimized, specs["duration"])
        
    elif specs["editing_style"] == "instant_impact":

        optimized = extract_most_impactful_moment(optimized)
    
    # Broadcast safe levels for TV
    if specs.get("safe_levels", False):
        optimized = broadcast_safe_levels(optimized)
    
    return optimized

def smart_trim_to_duration(clip, target_duration, style):
    """Intelligent trimming preserving narrative flow"""
    
    if style == "rapid_highlights":
        # TV spot: Select highest-impact moments
        impactful_moments = detect_impactful_segments(clip)
        total_duration = sum(m["duration"] for m in impactful_moments[:8])
        
        selected_clips = []
        current_time = 0
        
        for moment in impactful_moments[:8]:
            moment_clip = clip.subclip(moment["start"], moment["end"])
            trim_duration = (moment["duration"] / total_duration) * target_duration * 0.9
            selected_clips.append(moment_clip.subclip(0, trim_duration))
            current_time += trim_duration
            
            if current_time >= target_duration:
                break
        
        # Add quick cuts between moments
        return concatenate_videoclips(selected_clips, method="compose")
        
    elif style == "attention_grab":
        # Social media: Front-load most engaging content
        engaging_start = find_most_engaging_start(clip, target_duration * 1.5)
        return clip.subclip(engaging_start, engaging_start + target_duration)
    
    # Default: Proportional trimming
    scale_factor = target_duration / clip.duration
    return clip.subclip(0, target_duration)

def calculate_vertical_crop_center(clip, specs):
    """AI-guided focal point detection for vertical cropping"""
    
    # Use CLIP to identify important regions (faces, action)
    focal_queries = [
        "close-up of person's face",
        "action scene with movement", 
        "important object or device",
        "dramatic gesture or pose"
    ]
    
    # Sample frames and analyze
    sample_frames = sample_frames_uniformly(clip, num_samples=10)
    focal_points = []
    
    for frame_data in sample_frames:
        frame = frame_data["frame"]
        timestamp = frame_data["timestamp"]
        
        # CLIP analysis (simplified - would use full model in production)
        # For demo, use face detection as proxy
        faces = detect_faces(frame)
        
        if faces:
            # Average face center Y position
            avg_y = np.mean([face["center_y"] for face in faces])
            focal_points.append({
                "timestamp": timestamp,
                "focal_y": avg_y,
                "confidence": len(faces) / 10.0  # Normalized confidence
            })
    
    # Weighted average focal point
    if focal_points:
        weights = [p["confidence"] for p in focal_points]
        weighted_y = np.average([p["focal_y"] for p in focal_points], weights=weights)
        return int(weighted_y)
    
    # Default to frame center
    w, h = clip.size
    return w // 2

def parse_srt_file(srt_path):
    """Parse SRT subtitle file with timestamps"""
    subtitles = []
    
    with open(srt_path, 'r', encoding='utf-8') as f:
        content = f.read().strip()
        entries = content.split('\n\n')
        
        for entry in entries:
            lines = entry.strip().split('\n')
            if len(lines) >= 3:
                # Parse timestamp line (format: HH:MM:SS,mmm --> HH:MM:SS,mmm)
                timestamp_line = lines[1]
                time_parts = timestamp_line.split(' --> ')
                
                if len(time_parts) == 2:
                    start_time = srt_time_to_seconds(time_parts[0])
                    end_time = srt_time_to_seconds(time_parts[1])
                    
                    # Join subtitle text (multiple lines possible)
                    text = ' '.join(lines[2:])
                    
                    subtitles.append({
                        "index": len(subtitles) + 1,
                        "start_time": start_time,
                        "end_time": end_time,
                        "duration": end_time - start_time,
                        "text": text.strip()
                    })
    
    return subtitles

def srt_time_to_seconds(srt_time):
    """Convert SRT timestamp to seconds"""
    # Format: HH:MM:SS,mmm
    parts = srt_time.replace(',', '.').split(':')
    hours = int(parts[0])
    minutes = int(parts[1])
    seconds = float(parts[2])
    
    return hours * 3600 + minutes * 60 + seconds

            # Aggregate cross-platform metrics

            self.test_results[campaign_id]["daily_metrics"][f"day_{day+1}"] = aggregated
            
            # Early stopping if significant winner emerges
            if day > 3:  # After minimum 3 days
                significance = self.test_early_significance(aggregated)
                if significance["clear_winner"]:
                    print(f"Early winner detected: {significance['winner']} (p={significance['p_value']:.3f})")
                    break
            
            # Wait 24 hours between collections
            time.sleep(24 * 3600)
    
    def aggregate_daily_metrics(self, platform_data):
        """Combine metrics from all platforms into unified view"""
        
        unified_metrics = {
            "total_impressions": 0,
            "total_views": 0,
            "total_engagement": 0,
            "avg_view_duration": 0,
            "completion_rates": {},
            "click_through_rates": {},
            "platform_breakdown": {}
        }
        
        total_weighted_duration = 0
        weighted_duration_sum = 0
        
        for platform, data in platform_data.items():
            # Normalize metrics to common scale
            unified_metrics["total_impressions"] += data.get("impressions", 0)
            unified_metrics["total_views"] += data.get("video_views", 0)
            
            engagement = (data.get("likes", 0) + 
                         data.get("shares", 0) + 
                         data.get("comments", 0))
            unified_metrics["total_engagement"] += engagement
            
            # Weighted average view duration
            platform_duration = data.get("avg_view_duration", 0)
            platform_views = data.get("video_starts", 1)
            weighted_duration_sum += platform_duration * platform_views
            total_weighted_duration += platform_views
            
            unified_metrics["platform_breakdown"][platform] = data
        
        if total_weighted_duration > 0:
            unified_metrics["avg_view_duration"] = weighted_duration_sum / total_weighted_duration
        
        # Calculate rates for each variant
        for variant_name in self.test_results[campaign_id]["variants"].keys():
            variant_data = self.extract_variant_metrics(platform_data, variant_name)
            
            if variant_data["impressions"] > 0:
                unified_metrics["completion_rates"][variant_name] = (
                    variant_data["completed_views"] / variant_data["video_starts"]
                )
                unified_metrics["click_through_rates"][variant_name] = (
                    variant_data["clicks"] / variant_data["impressions"]
                )
        
        return unified_metrics
    
    def test_early_significance(self, metrics_data):
        """Statistical testing for early winner detection"""
        
        variants = list(metrics_data["completion_rates"].keys())
        if len(variants) < 2:
            return {"clear_winner": False}
        
        # Extract completion rates
        completion_rates = [metrics_data["completion_rates"][v] for v in variants]
        views = [self.get_variant_views(v) for v in variants]  # Sample sizes
        
        # One-way ANOVA to test if differences are significant
        f_stat, p_value = stats.f_oneway(*[
            np.random.normal(rate, 0.05, int(view_count * 0.1))  # Bootstrap samples
            for rate, view_count in zip(completion_rates, views)
        ])
        
        # Identify best performer
        best_variant = variants[np.argmax(completion_rates)]
        best_rate = max(completion_rates)
        
        # Bonferroni correction for multiple comparisons
        alpha = 0.05 / len(variants)
        clear_winner = p_value < alpha and best_rate > np.mean(completion_rates) + 0.1
        
        return {
            "clear_winner": clear_winner,
            "best_variant": best_variant,
            "best_performance": best_rate,
            "p_value": p_value,
            "statistical_power": calculate_statistical_power(views, completion_rates)
        }
    
    def generate_final_recommendations(self, campaign_id):
        """Comprehensive analysis and recommendations"""
        
        results = self.test_results[campaign_id]
        all_metrics = []
        
        # Aggregate all daily data
        for day_data in results["daily_metrics"].values():
            all_metrics.extend(self.flatten_metrics(day_data))
        
        # Create performance DataFrame
        df = pd.DataFrame(all_metrics)
        
        # Key performance indicators by variant
        performance_summary = df.groupby("variant").agg({
            "completion_rate": ["mean", "std", np.size],
            "click_through_rate": ["mean", "std", np.size],
            "engagement_rate": ["mean", "std", np.size],
            "impressions": "sum",
            "video_views": "sum"
        }).round(4)
        
        # Statistical significance testing
        significance_results = self.perform_statistical_tests(df)
        
        # Audience insights
        demographic_analysis = self.analyze_demographic_performance(df)
        
        # ROI calculations
        cost_data = self.calculate_campaign_costs(campaign_id)
        roi_analysis = self.calculate_roi_by_variant(performance_summary, cost_data)
        
        # Generate recommendations
        recommendations = self.create_actionable_recommendations(
            performance_summary, 
            significance_results, 
            demographic_analysis, 
            roi_analysis
        )
        
        # Final report
        final_report = {
            "campaign_id": campaign_id,
            "test_duration": (results["end_time"] - results["start_time"]).days,
            "total_impressions": int(df["impressions"].sum()),
            "total_engagement": int(df["total_engagement"].sum()),
            "performance_by_variant": performance_summary.to_dict(),
            "statistical_significance": significance_results,
            "audience_insights": demographic_analysis,
            "roi_analysis": roi_analysis,
            "recommendations": recommendations,
            "best_variant": self.determine_winning_variant(performance_summary, significance_results),
            "confidence_level": self.calculate_recommendation_confidence(performance_summary)
        }
        
        return final_report
    
    def determine_winning_variant(self, performance, significance):
        """Select optimal variant based on multiple criteria"""
        
        # Composite scoring: 40% completion, 30% CTR, 20% engagement, 10% statistical confidence
        variant_scores = {}
        
        for variant in performance.index:
            completion_score = performance.loc[variant, ("completion_rate", "mean")]
            ctr_score = performance.loc[variant, ("click_through_rate", "mean")]
            engagement_score = performance.loc[variant, ("engagement_rate", "mean")]
            
            # Statistical confidence bonus
            p_value = significance.get(variant, {}).get("p_value", 1.0)
            confidence_bonus = 1.0 - min(p_value, 0.05) / 0.05  # 0-1 scale
            
            composite_score = (
                completion_score * 0.40 +
                ctr_score * 0.30 +
                engagement_score * 0.20 +
                confidence_bonus * 0.10
            )
            
            variant_scores[variant] = composite_score
        
        winner = max(variant_scores, key=variant_scores.get)
        winner_score = variant_scores[winner]
        
        return {
            "winning_variant": winner,
            "composite_score": winner_score,
            "score_breakdown": variant_scores,
            "performance_margin": winner_score - np.mean(list(variant_scores.values()))
        }

def process_music_for_trailer(music, timeline, sr):
    """Apply trailer-specific music processing"""
    
    # Multiband compression for cinematic sound
    low_freq, mid_freq, high_freq = 250, 4000, 8000
    
    # Create crossover filters
    sos_low = signal.butter(6, low_freq, btype='low', fs=sr, output='sos')
    sos_mid = signal.butter(6, [low_freq, mid_freq], btype='band', fs=sr, output='sos')
    sos_high = signal.butter(6, high_freq, btype='high', fs=sr, output='sos')
    
    # Split into frequency bands
    low_band = signal.sosfilt(sos_low, music, axis=1)
    mid_band = signal.sosfilt(sos_mid, music, axis=1)
    high_band = signal.sosfilt(sos_high, music, axis=1)
    
    # Apply genre-specific compression
    low_compressed = apply_multiband_compression(low_band, threshold=-25, ratio=4, sr=sr)
    mid_compressed = apply_multiband_compression(mid_band, threshold=-20, ratio=6, sr=sr)
    high_compressed = apply_multiband_compression(high_band, threshold=-18, ratio=8, sr=sr)
    
    # Recombine bands
    recombined = low_compressed + mid_compressed + high_compressed

    reverb = apply_convolution_reverb(recombined, "cinematic_hall_ir.wav", reverb_tail)

def apply_multiband_compression(audio, threshold, ratio, sr):
    """Apply frequency-specific compression"""
    
    # Convert to dB
    audio_db = 20 * np.log10(np.abs(audio) + 1e-10)
    
    # Gain reduction calculation
    over_threshold = audio_db > threshold
    gain_reduction = np.zeros_like(audio_db)
    
    gain_reduction[over_threshold] = (
        (audio_db[over_threshold] - threshold) * (1 - 1/ratio)
    )
    
    # Apply attack/release smoothing (10ms attack, 100ms release)
    attack_samples = int(0.01 * sr)
    release_samples = int(0.1 * sr)
    
    for channel in range(audio.shape[0]):
        gain_reduction[channel] = smooth_gain_reduction(
            gain_reduction[channel], attack_samples, release_samples
        )
    
    # Convert back to linear gain
    gain_linear = 10 ** (gain_reduction / 20)
    compressed = audio * gain_linear
    
    return compressed

def intelligent_audio_ducking(music, dialogue, sfx, sr):
    """Smart ducking based on content priority"""
    
    # Analyze dialogue presence (primary element)
    dialogue_rms = librosa.feature.rms(y=dialogue, hop_length=512)[0]
    dialogue_db = 20 * np.log10(dialogue_rms + 1e-10)
    dialogue_threshold = np.percentile(dialogue_db, 75)  # Active speech threshold
    
    # Create time-smoothed dialogue envelope
    dialogue_envelope = np.convolve(
        (dialogue_db > dialogue_threshold).astype(float),
        np.ones(1024)/1024,  # 21ms smoothing window at 48kHz
        mode='same'
    )
    
    # Calculate ducking curves (6dB music reduction, 3dB SFX reduction during speech)
    music_ducking = 1.0 - (dialogue_envelope * 0.5)  # Max 6dB reduction
    sfx_ducking = 1.0 - (dialogue_envelope * 0.25)    # Max 3dB reduction
    
    # Frequency-specific ducking (protect dialogue range 200-4000Hz)
    music_ducked = frequency_aware_ducking(music, music_ducking, dialogue_envelope, sr)
    sfx_ducked = sfx * sfx_ducking
    
    # Mix levels (dialogue dominant, music support, sfx accent)
    dialogue_level = 0.0  # Dialogue at 0 dBFS reference
    music_level = -12     # Music at -12 dBFS
    sfx_level = -18       # SFX at -18 dBFS
    
    # Apply final mixing
    final_mix = (
        dialogue * dialogue_level +
        music_ducked * music_level +
        sfx_ducked * sfx_level
    )
    
    # Stereo enhancement
    final_mix = stereo_imaging(final_mix)
    
    return final_mix

def master_for_broadcast(audio, sr, target_lufs=-14, max_peak=-1):
    """Apply broadcast-standard loudness normalization"""
    
    # Measure integrated LUFS
    current_lufs = measure_lufs(audio, sr)
    
    # Calculate required gain
    gain_db = target_lufs - current_lufs
    gain_linear = 10 ** (gain_db / 20)
    
    # Apply gain
    normalized = audio * gain_linear
    
    # True peak limiting
    peak_limited = true_peak_limiter(normalized, max_peak, sr)
    
    # Final safety clip
    final_mix = np.clip(peak_limited, -1.0, 1.0)
    
    # Verify final loudness
    final_lufs = measure_lufs(final_mix, sr)
    final_peak = 20 * np.log10(np.max(np.abs(final_mix)))
    
    print(f"Mastered: {final_lufs:.1f} LUFS, {final_peak:.1f} dBTP")
    
    return final_mix

def measure_lufs(audio, sr, block_size=400):  # 400ms blocks per EBU R128
    """Calculate Loudness Units relative to Full Scale (ITU-R BS.1770)"""
    
    # K-weighting filter (high shelf + high pass)
    # High shelf filter (4kHz)
    b_hf, a_hf = signal.iirfilter(
        2, 7380, btype='high', analog=True, ftype='butter'
    )
    b_hf, a_hf = signal.bilinear(b_hf, a_hf, fs=sr)
    
    # High pass filter (50Hz)
    b_hp, a_hp = signal.iirfilter(
        2, 50, btype='high', analog=True, ftype='butter'
    )
    b_hp, a_hp = signal.bilinear(b_hp, a_hp, fs=sr)
    
    # Apply weighting filters
    weighted_low = signal.filtfilt(b_hp, a_hp, audio, axis=1)
    weighted = signal.filtfilt(b_hf, a_hf, weighted_low, axis=1)
    
    # Calculate mean square
    mean_square = np.mean(weighted**2, axis=0)
    
    # Gated integration (absolute threshold -70 LUFS)
    block_samples = int(block_size * sr / 1000)
    blocks = []
    
    for i in range(0, len(mean_square) - block_samples, block_samples // 4):
        block_power = np.mean(mean_square[i:i+block_samples])
        if block_power > 0:
            block_lufs = -0.691 + 10 * np.log10(block_power)  # 0.691 = 10*log10(2**23)
            
            if block_lufs > -70:  # Absolute gating
                blocks.append(block_lufs)
    
    if not blocks:
        return -70.0
    
    # Relative gating (-10 LUFS below ungated mean)
    ungated_mean = np.mean(blocks)
    relative_threshold = ungated_mean - 10
    gated_blocks = [b for b in blocks if b > relative_threshold]
    
    if gated_blocks:
        integrated_lufs = np.mean(gated_blocks)
    else:
        integrated_lufs = ungated_mean
    
    return integrated_lufs

def apply_platform_optimizations(video_clip, specs):
    """Apply platform-specific video processing"""
    
    # Duration adaptation
    if video_clip.duration > specs["duration"]:
        # Smart trimming - preserve key moments
        video_clip = smart_trim_to_duration(video_clip, specs["duration"], specs["editing_style"])
    elif video_clip.duration < specs["duration"]:
        # Loop or extend final moment

    # Resolution and aspect ratio
    target_width, target_height = specs["resolution"]
    
    if specs["aspect_ratio"] < 1.0:  # Vertical format
        # Smart cropping for vertical platforms
        resized = video_clip.resize(height=target_height)
        w, h = resized.size

        cropped = resized.crop(x1=crop_x - target_width//2, y1=0,

        optimized = cropped.resize((target_width, target_height))
        
        # Add social media overlays if specified
        if "social_overlays" in specs:
            optimized = add_platform_branding(optimized, specs["platform_branding"])
            
    else:  # Horizontal format
        # Standard letterboxing or pillarboxing
        resized = video_clip.resize(width=target_width)
        
        if resized.h < target_height:
            # Letterbox (black bars top/bottom)
            optimized = resized.on_color(
                size=(target_width, target_height), 
                color=(0, 0, 0), col_opacity=1
            )
        else:
            # Pillarbox (black bars sides)
            w, h = resized.size
            crop_y = h//2 - target_height//2
            optimized = resized.crop(y1=crop_y, y2=crop_y + target_height)
            optimized = optimized.resize((target_width, target_height))
    
    # Frame rate conversion
    if abs(optimized.fps - specs["framerate"]) > 0.1:
        optimized = optimized.set_fps(specs["framerate"])
    
    # Color space conversion
    if specs["color_space"] == "BT.709":  # Standard for broadcast/web
        # Apply Rec.709 color matrix if source is film/DCI-P3
        pass  # Implementation depends on source color space
    elif specs["color_space"] == "BT.2020":  # Theatrical 4K
        # Preserve wide color gamut
        pass
    
    # Platform-specific effects
    if specs["editing_style"] == "rapid_highlights":
        # TV spot: Faster pacing, higher contrast
        optimized = optimized.fx(lambda c: c.adjust_contrast(1.2))
        optimized = optimized.fx(lambda c: c.speedx(1.2))  # 20% faster

def smart_trim_to_duration(clip, target_duration, style):
    """Intelligent trimming preserving narrative flow"""
    
    if style == "rapid_highlights":
        # TV spot: Select highest-impact moments
        impactful_moments = detect_impactful_segments(clip)
        total_duration = sum(m["duration"] for m in impactful_moments[:8])
        
        selected_clips = []
        current_time = 0
        
        for moment in impactful_moments[:8]:
            moment_clip = clip.subclip(moment["start"], moment["end"])
            trim_duration = (moment["duration"] / total_duration) * target_duration * 0.9
            selected_clips.append(moment_clip.subclip(0, trim_duration))
            current_time += trim_duration
            
            if current_time >= target_duration:
                break
        
        # Add quick cuts between moments
        return concatenate_videoclips(selected_clips, method="compose")
        
    elif style == "attention_grab":
        # Social media: Front-load most engaging content
        engaging_start = find_most_engaging_start(clip, target_duration * 1.5)
        return clip.subclip(engaging_start, engaging_start + target_duration)
    
    # Default: Proportional trimming
    scale_factor = target_duration / clip.duration
    return clip.subclip(0, target_duration)

def calculate_vertical_crop_center(clip, specs):
    """AI-guided focal point detection for vertical cropping"""

    focal_queries = [

        "important object or device",
        "dramatic gesture or pose"
    ]
    
    # Sample frames and analyze
    sample_frames = sample_frames_uniformly(clip, num_samples=10)
    focal_points = []
    
    for frame_data in sample_frames:
        frame = frame_data["frame"]
        timestamp = frame_data["timestamp"]
        
        # CLIP analysis (simplified - would use full model in production)
        # For demo, use face detection as proxy
        faces = detect_faces(frame)
        
        if faces:
            # Average face center Y position
            avg_y = np.mean([face["center_y"] for face in faces])
            focal_points.append({
                "timestamp": timestamp,
                "focal_y": avg_y,
                "confidence": len(faces) / 10.0  # Normalized confidence
            })
    
    # Weighted average focal point
    if focal_points:
        weights = [p["confidence"] for p in focal_points]
        weighted_y = np.average([p["focal_y"] for p in focal_points], weights=weights)
        return int(weighted_y)
    
    # Default to frame center
    w, h = clip.size
    return w // 2

def parse_srt_file(srt_path):
    """Parse SRT subtitle file with timestamps"""
    subtitles = []
    
    with open(srt_path, 'r', encoding='utf-8') as f:
        content = f.read().strip()
        entries = content.split('\n\n')
        
        for entry in entries:
            lines = entry.strip().split('\n')
            if len(lines) >= 3:
                # Parse timestamp line (format: HH:MM:SS,mmm --> HH:MM:SS,mmm)
                timestamp_line = lines[1]
                time_parts = timestamp_line.split(' --> ')
                
                if len(time_parts) == 2:
                    start_time = srt_time_to_seconds(time_parts[0])
                    end_time = srt_time_to_seconds(time_parts[1])
                    
                    # Join subtitle text (multiple lines possible)
                    text = ' '.join(lines[2:])
                    
                    subtitles.append({
                        "index": len(subtitles) + 1,
                        "start_time": start_time,
                        "end_time": end_time,
                        "duration": end_time - start_time,
                        "text": text.strip()
                    })
    
    return subtitles

def srt_time_to_seconds(srt_time):
    """Convert SRT timestamp to seconds"""
    # Format: HH:MM:SS,mmm
    parts = srt_time.replace(',', '.').split(':')
    hours = int(parts[0])
    minutes = int(parts[1])
    seconds = float(parts[2])
    
    return hours * 3600 + minutes * 60 + seconds

            day_data = {}
            
            for platform in self.platform_apis.keys():
                platform_metrics = self.platform_apis[platform].collect_metrics(campaign_id)
                day_data[platform] = platform_metrics
            
            # Aggregate cross-platform metrics
            aggregated = self.aggregate_daily_metrics(day_data)
            self.test_results[campaign_id]["daily_metrics"][f"day_{day+1}"] = aggregated
            
            # Early stopping if significant winner emerges
            if day > 3:  # After minimum 3 days
                significance = self.test_early_significance(aggregated)
                if significance["clear_winner"]:
                    print(f"Early winner detected: {significance['winner']} (p={significance['p_value']:.3f})")
                    break
            
            # Wait 24 hours between collections
            time.sleep(24 * 3600)
    
    def aggregate_daily_metrics(self, platform_data):
        """Combine metrics from all platforms into unified view"""
        
        unified_metrics = {
            "total_impressions": 0,
            "total_views": 0,
            "total_engagement": 0,
            "avg_view_duration": 0,
            "completion_rates": {},
            "click_through_rates": {},
            "platform_breakdown": {}
        }
        
        total_weighted_duration = 0
        weighted_duration_sum = 0
        
        for platform, data in platform_data.items():
            # Normalize metrics to common scale
            unified_metrics["total_impressions"] += data.get("impressions", 0)
            unified_metrics["total_views"] += data.get("video_views", 0)
            
            engagement = (data.get("likes", 0) + 
                         data.get("shares", 0) + 
                         data.get("comments", 0))
            unified_metrics["total_engagement"] += engagement
            
            # Weighted average view duration
            platform_duration = data.get("avg_view_duration", 0)
            platform_views = data.get("video_starts", 1)
            weighted_duration_sum += platform_duration * platform_views
            total_weighted_duration += platform_views
            
            unified_metrics["platform_breakdown"][platform] = data
        
        if total_weighted_duration > 0:
            unified_metrics["avg_view_duration"] = weighted_duration_sum / total_weighted_duration
        
        # Calculate rates for each variant
        for variant_name in self.test_results[campaign_id]["variants"].keys():
            variant_data = self.extract_variant_metrics(platform_data, variant_name)
            
            if variant_data["impressions"] > 0:
                unified_metrics["completion_rates"][variant_name] = (
                    variant_data["completed_views"] / variant_data["video_starts"]
                )
                unified_metrics["click_through_rates"][variant_name] = (
                    variant_data["clicks"] / variant_data["impressions"]
                )
        
        return unified_metrics
    
    def test_early_significance(self, metrics_data):
        """Statistical testing for early winner detection"""
        
        variants = list(metrics_data["completion_rates"].keys())
        if len(variants) < 2:
            return {"clear_winner": False}
        
        # Extract completion rates
        completion_rates = [metrics_data["completion_rates"][v] for v in variants]
        views = [self.get_variant_views(v) for v in variants]  # Sample sizes
        
        # One-way ANOVA to test if differences are significant
        f_stat, p_value = stats.f_oneway(*[
            np.random.normal(rate, 0.05, int(view_count * 0.1))  # Bootstrap samples
            for rate, view_count in zip(completion_rates, views)
        ])
        
        # Identify best performer
        best_variant = variants[np.argmax(completion_rates)]
        best_rate = max(completion_rates)
        
        # Bonferroni correction for multiple comparisons
        alpha = 0.05 / len(variants)
        clear_winner = p_value < alpha and best_rate > np.mean(completion_rates) + 0.1
        
        return {
            "clear_winner": clear_winner,
            "best_variant": best_variant,
            "best_performance": best_rate,
            "p_value": p_value,
            "statistical_power": calculate_statistical_power(views, completion_rates)
        }
    
    def generate_final_recommendations(self, campaign_id):
        """Comprehensive analysis and recommendations"""
        
        results = self.test_results[campaign_id]
        all_metrics = []
        
        # Aggregate all daily data
        for day_data in results["daily_metrics"].values():
            all_metrics.extend(self.flatten_metrics(day_data))
        
        # Create performance DataFrame
        df = pd.DataFrame(all_metrics)
        
        # Key performance indicators by variant
        performance_summary = df.groupby("variant").agg({
            "completion_rate": ["mean", "std", np.size],
            "click_through_rate": ["mean", "std", np.size],
            "engagement_rate": ["mean", "std", np.size],
            "impressions": "sum",
            "video_views": "sum"
        }).round(4)
        
        # Statistical significance testing
        significance_results = self.perform_statistical_tests(df)
        
        # Audience insights
        demographic_analysis = self.analyze_demographic_performance(df)
        
        # ROI calculations
        cost_data = self.calculate_campaign_costs(campaign_id)
        roi_analysis = self.calculate_roi_by_variant(performance_summary, cost_data)
        
        # Generate recommendations
        recommendations = self.create_actionable_recommendations(
            performance_summary, 
            significance_results, 
            demographic_analysis, 
            roi_analysis
        )
        
        # Final report
        final_report = {
            "campaign_id": campaign_id,
            "test_duration": (results["end_time"] - results["start_time"]).days,
            "total_impressions": int(df["impressions"].sum()),
            "total_engagement": int(df["total_engagement"].sum()),
            "performance_by_variant": performance_summary.to_dict(),
            "statistical_significance": significance_results,
            "audience_insights": demographic_analysis,
            "roi_analysis": roi_analysis,
            "recommendations": recommendations,
            "best_variant": self.determine_winning_variant(performance_summary, significance_results),
            "confidence_level": self.calculate_recommendation_confidence(performance_summary)
        }
        
        return final_report
    
    def determine_winning_variant(self, performance, significance):
        """Select optimal variant based on multiple criteria"""
        
        # Composite scoring: 40% completion, 30% CTR, 20% engagement, 10% statistical confidence
        variant_scores = {}
        
        for variant in performance.index:
            completion_score = performance.loc[variant, ("completion_rate", "mean")]
            ctr_score = performance.loc[variant, ("click_through_rate", "mean")]
            engagement_score = performance.loc[variant, ("engagement_rate", "mean")]
            
            # Statistical confidence bonus
            p_value = significance.get(variant, {}).get("p_value", 1.0)
            confidence_bonus = 1.0 - min(p_value, 0.05) / 0.05  # 0-1 scale
            
            composite_score = (
                completion_score * 0.40 +
                ctr_score * 0.30 +
                engagement_score * 0.20 +
                confidence_bonus * 0.10
            )
            
            variant_scores[variant] = composite_score
        
        winner = max(variant_scores, key=variant_scores.get)
        winner_score = variant_scores[winner]
        
        return {
            "winning_variant": winner,
            "composite_score": winner_score,
            "score_breakdown": variant_scores,
            "performance_margin": winner_score - np.mean(list(variant_scores.values()))
        }

        creative_edit = self.video_editor.assemble_creative_cut(
            sync_timeline, music_assets, title_sequence
        )
        
        # Creative review points
        creative_decisions = {
            "music_sync_accuracy": self.evaluate_sync_quality(sync_timeline),
            "pacing_effectiveness": self.assess_pacing_flow(creative_edit),
            "title_impact": self.measure_title_readability(title_sequence),
            "overall_mood": self.classify_trailer_mood(creative_edit)
        }
        
        return {
            "creative_edit": creative_edit,
            "music_integration": music_analysis,
            "title_cards": title_sequence,
            "creative_decisions": creative_decisions,
            "mood_profile": self.analyze_emotional_arc(creative_edit)
        }
    
    def phase_4_technical_production(self, creative_assets):
        """Technical finishing and quality optimization"""
        
        # Audio mastering
        mastered_audio = self.audio_processor.master_for_distribution(
            creative_assets["creative_edit"]
        )
        
        # Video technical specifications
        technical_video = self.video_editor.apply_technical_standards(
            creative_assets["creative_edit"]
        )
        
        # Quality control checks
        technical_specs = {
            "video": self.validate_video_standards(technical_video),
            "audio": self.validate_audio_standards(mastered_audio),
            "compliance": self.check_broadcast_compliance(technical_video, mastered_audio),
            "file_sizes": self.measure_output_sizes(technical_video, mastered_audio)
        }
        
        # Final master creation
        master_package = {
            "video_master": technical_video,
            "audio_master": mastered_audio,
            "technical_report": technical_specs,
            "render_metadata": self.document_render_settings()
        }
        
        return master_package
    
    def phase_5_platform_versions(self, master_package):
        """Generate platform-optimized versions"""
        
        platform_configs = self.config["platform_versions"]
        versions = {}
        
        for platform, config in platform_configs.items():
            self.logger.info(f"Creating {platform} version...")
            
            # Adapt master for platform
            platform_video = self.adapt_for_platform(
                master_package["video_master"], config
            )
            
            platform_audio = self.adapt_audio_for_platform(
                master_package["audio_master"], config
            )
            
            # Add platform-specific elements
            final_version = self.add_platform_branding(
                platform_video.set_audio(platform_audio), config
            )
            
            # Export with platform specifications
            output_path = f"trailer_{platform}_final.mp4"
            self.video_editor.export_final_version(
                final_version, output_path, config["export_settings"]
            )
            
            versions[platform] = {
                "path": output_path,
                "duration": config["duration"],
                "resolution": config["resolution"],
                "technical_specs": config["export_settings"],
                "file_size": Path(output_path).stat().st_size,
                "quality_metrics": self.measure_version_quality(final_version)
            }
        
        return versions
    
    def phase_6_quality_assurance(self, platform_versions):
        """Comprehensive quality validation"""
        
        qa_results = {}
        
        for platform, version in platform_versions.items():

            video_path = version["path"]
            
            # Technical quality checks
            technical_qa = {
                "video_validation": self.validate_video_file(video_path),
                "audio_validation": self.validate_audio_track(video_path),
                "format_compliance": self.check_format_standards(video_path, platform),
                "performance_metrics": self.measure_playback_performance(video_path)

            # Aesthetic quality assessment
            aesthetic_qa = {
                "pacing_analysis": self.analyze_pacing_quality(video_path),
                "visual_consistency": self.check_color_consistency(video_path),
                "audio_balance": self.assess_audio_mix(video_path),
                "title_readability": self.evaluate_text_elements(video_path)
            }
            
            # Automated accessibility checks
            accessibility_qa = {
                "contrast_ratios": self.measure_text_contrast(video_path),
                "audio_description": self.check_audio_cues(video_path),
                "subtitle_sync": self.validate_subtitle_timing(video_path) if platform != "theatrical" else None
            }
            
            qa_results[platform] = {
                "technical": technical_qa,
                "aesthetic": aesthetic_qa,
                "accessibility": accessibility_qa,
                "overall_score": self.calculate_qa_score(technical_qa, aesthetic_qa, accessibility_qa),
                "issues": self.compile_issue_list(technical_qa, aesthetic_qa, accessibility_qa),
                "recommendations": self.generate_qa_fixes(technical_qa, aesthetic_qa, accessibility_qa)
            }
        
        # Overall pipeline quality summary
        pipeline_quality = {
            "consistency_across_versions": self.measure_version_consistency(platform_versions),
            "brand_compliance": self.validate_brand_standards(platform_versions),
            "legal_compliance": self.check_content_clearance(platform_versions),
            "overall_pipeline_score": self.calculate_pipeline_quality(qa_results)
        }

            "platform_qa": qa_results,
            "pipeline_quality": pipeline_quality,
            "final_validation": self.generate_qa_summary(qa_results, pipeline_quality)
        }

        return f"{size_mb:.1f} MB"
