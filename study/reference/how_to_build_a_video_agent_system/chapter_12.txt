```python
# Core Cinematography Agent Architecture
import bpy
import bmesh
import mathutils
from mathutils import Vector, Euler
import numpy as np
import clip
import torch

class CinematographyAgent:
    def __init__(self):
        self.device = "cuda" if torch.cuda.is_available() else "cpu"
        self.clip_model, self.clip_preprocess = clip.load('ViT-B/32', self.device)

    def setup_camera_rig(self, scene_bounds: dict, shot_type: str):
        """Generate camera with precise parametric control"""
        # Create camera with low-level API access
        camera_data = bpy.data.cameras.new(name="CineCamera")
        camera_object = bpy.data.objects.new("CineCamera", camera_data)
        bpy.context.collection.objects.link(camera_object)

        # Configure lens properties based on shot type
        if shot_type == "wide_master":
            camera_data.lens = 35  # 35mm lens
            camera_data.clip_end = 1000
        elif shot_type == "close_up":
            camera_data.lens = 85  # Portrait lens
            camera_data.dof.use_dof = True
            camera_data.dof.aperture_fstop = 8

        return camera_object

    def generate_camera_path(self, camera_obj, movement_type: str, duration: int):
        """Generate smooth camera trajectories using spline interpolation"""
        if movement_type == "handheld":
            # Multi-layered Perlin noise for authentic operator movement
            for frame in range(1, duration + 1):
                bpy.context.scene.frame_set(frame)

                # Primary shake (3Hz)
                primary_offset = self._perlin_noise(frame * 3) * 0.02
                # Secondary tremor (8Hz) 
                secondary_offset = self._perlin_noise(frame * 8) * 0.015
                # Breathing drift (2Hz)
                breathing_offset = self._perlin_noise(frame * 2) * 0.01

                total_offset = Vector((
                    primary_offset + secondary_offset,
                    breathing_offset,
                    secondary_offset * 0.5
                ))

                camera_obj.location += total_offset
                camera_obj.keyframe_insert(data_path="location")

        elif movement_type == "orbital":
            # 270° orbit with dynamic radius variation
            center = Vector((0, 0, 0))
            for frame in range(1, duration + 1):
                angle = (frame / duration) * (270 * np.pi / 180)
                radius = 14 + 2 * np.sin(angle * 5)  # 12-16 feet variance

                x = center.x + radius * np.cos(angle)
                y = center.y + radius * np.sin(angle)
                z = 5 + 3 * np.sin(angle * 2)  # Height variation

                camera_obj.location = Vector((x, y, z))
                camera_obj.keyframe_insert(data_path="location")

    def analyze_composition(self, rendered_image_path: str):
        """Use CLIP for composition analysis"""
        from PIL import Image

        image = Image.open(rendered_image_path)
        image_input = self.clip_preprocess(image).unsqueeze(0).to(self.device)

        # Define cinematographic concepts for analysis
        cinematography_concepts = [
            "rule of thirds composition",
            "symmetrical framing", 
            "leading lines composition",
            "depth of field photography",
            "low angle dramatic shot",
            "high angle overhead shot",
            "close up portrait",
            "wide establishing shot"
        ]

        text_inputs = torch.cat([
            clip.tokenize(f"a photo with {concept}") 
            for concept in cinematography_concepts
        ]).to(self.device)

        with torch.no_grad():
            image_features = self.clip_model.encode_image(image_input)
            text_features = self.clip_model.encode_text(text_inputs)

            # Calculate cosine similarity 
            image_features /= image_features.norm(dim=-1, keepdim=True)
            text_features /= text_features.norm(dim=-1, keepdim=True)
            similarity = (100 * image_features @ text_features.T).softmax(dim=-1)

        return {concept: float(sim) for concept, sim in 
                zip(cinematography_concepts, similarity[0])}

    def _perlin_noise(self, t):
        """Simplified Perlin noise implementation"""
        # Placeholder for Perlin noise function
        return np.sin(t) * 0.1
```

**Cons**

```python
# Scene Management with BlenderProc Integration
import blenderproc as bproc
import numpy as np
from typing import Dict, List
import json

class SceneManager:
    def __init__(self):
        bproc.init()
        self.scene_config = {}

    def setup_victorian_mansion_scene(self, config_path: str):
        """Load and configure Victorian mansion environment"""

        # Load scene configuration
        with open(config_path, 'r') as f:
            self.scene_config = json.load(f)

        # Load 3D-FRONT scene with automatic lighting
        mansion_path = self.scene_config['mansion_model_path']
        objs = bproc.loader.load_3d_front(
            json_path=mansion_path,
            future_model_path=self.scene_config['furniture_path']
        )

        # Set up materials with PBR properties
        self.configure_pbr_materials(objs)

        # Configure procedural lighting for Victorian interior
        self.setup_procedural_lighting()

        return objs

    def configure_pbr_materials(self, objects: List):
        """Configure physically-based rendering materials"""

        for obj in objects:
            if obj.get_mesh() is not None:
                # Get or create material
                materials = obj.get_materials()
                if not materials:
                    mat = bproc.material.create("PBRMaterial")
                    obj.set_material(0, mat)
                    materials = [mat]

                # Configure PBR properties based on object type
                for mat in materials:
                    if "wood" in obj.get_name().lower():
                        # Wood material configuration
                        mat.set_principled_shader_value("Base Color", [0.4, 0.2, 0.1, 1.0])
                        mat.set_principled_shader_value("Roughness", 0.7)
                        mat.set_principled_shader_value("Specular", 0.3)

                    elif "fabric" in obj.get_name().lower():
                        # Fabric/upholstery material  
                        mat.set_principled_shader_value("Base Color", [0.3, 0.1, 0.1, 1.0])
                        mat.set_principled_shader_value("Roughness", 0.9)
                        mat.set_principled_shader_value("Specular", 0.1)

                    elif "metal" in obj.get_name().lower():
                        # Metallic surfaces (fixtures, handles)
                        mat.set_principled_shader_value("Base Color", [0.7, 0.7, 0.7, 1.0])
                        mat.set_principled_shader_value("Metallic", 0.9)
                        mat.set_principled_shader_value("Roughness", 0.2)

    def setup_procedural_lighting(self):
        """Configure automated lighting with domain randomization"""

        # Window light sampling based on scene configuration  
        window_locations = self.scene_config.get('window_positions', [])
        for i, pos in enumerate(window_locations):
            # Sample natural light with time-of-day variation
            energy = np.random.uniform(800, 1500)  # Overcast daylight range
            color_temp = np.random.uniform(5200, 5800)  # Cool daylight

            light = bproc.types.Light()
            light.set_type("AREA")
            light.set_location(pos)
            light.set_energy(energy)
            light.set_color(self._color_temp_to_rgb(color_temp))

        # Procedural interior lighting
        self._add_practical_lights()

    def _add_practical_lights(self):
        """Add motivated interior light sources"""

        # Chandelier configuration
        if 'chandelier_position' in self.scene_config:
            chandelier_pos = self.scene_config['chandelier_position']

            # Multiple bulbs with slight variation
            for i in range(8):  # 8-bulb chandelier
                angle = (i / 8) * 2 * np.pi
                offset_x = 0.5 * np.cos(angle)
                offset_y = 0.5 * np.sin(angle) 

                bulb_light = bproc.types.Light()
                bulb_light.set_type("POINT")
                bulb_light.set_location([
                    chandelier_pos[0] + offset_x,
                    chandelier_pos[1] + offset_y, 
                    chandelier_pos[2]
                ])
                bulb_light.set_energy(np.random.uniform(140, 160))
                bulb_light.set_color(self._color_temp_to_rgb(2800))

        # Fireplace with flickering animation
        if 'fireplace_position' in self.scene_config:
            fireplace_light = bproc.types.Light()
            fireplace_light.set_type("POINT")
            fireplace_light.set_location(self.scene_config['fireplace_position'])

            # Animate flickering
            for frame in range(1, 250):
                energy = np.random.uniform(400, 700)  
                fireplace_light.set_energy(energy, frame)

                # Color variation for flame effect
                temp = np.random.uniform(1900, 2200)
                fireplace_light.set_color(self._color_temp_to_rgb(temp), frame)

    def generate_camera_sampling_constraints(self) -> Dict:
        """Define camera sampling constraints for scene"""

        # Room boundaries for camera placement
        room_bounds = self.scene_config.get('room_dimensions', {
            'min_x': -10, 'max_x': 10,
            'min_y': -8, 'max_y': 8, 
            'min_z': 1, 'max_z': 7
        })

        # Obstacle avoidance for furniture
        furniture_bounds = [
            obj.get_bound_box() for obj in bproc.object.get_all_mesh_objects()
            if 'furniture' in obj.get_name().lower()
        ]

        return {
            'room_bounds': room_bounds,
            'obstacles': furniture_bounds,
            'min_subject_distance': 2,  # Minimum 2 feet from subjects
            'max_subject_distance': 20,  # Maximum 20 feet
            'preferred_height_range': (5, 7),  # Eye level range
            'excluded_zones': self.scene_config.get('no_camera_zones', [])
        }

    def _color_temp_to_rgb(self, temp_k: float) -> List[float]:
        """Convert color temperature to RGB values"""
        # Simplified blackbody radiation approximation
        if temp_k < 3000:
            return [1.0, 0.7, 0.4, 1.0]  # Warm tungsten
        elif temp_k < 4000:
            return [1.0, 0.9, 0.7, 1.0]  # Warm white
        elif temp_k < 5500:
            return [1.0, 1.0, 0.9, 1.0]  # Neutral white
        else:
            return [0.95, 0.95, 1.0, 1.0]  # Cool daylight
```

```typescript
// Real-time Preview System with React Three Fiber
import React, { useRef, useState, useEffect, Suspense } from 'react'
import { Canvas, useFrame, useThree } from '@react-three/fiber'
import { OrbitControls, PerspectiveCamera, Environment } from '@react-three/drei'
import * as THREE from 'three'

interface CameraPreviewProps {
  cameraPath: Array<{
    position: [number, number, number]
    rotation: [number, number, number] 
    focal_length: number
    frame: number
  }>
  currentFrame: number
  onFrameChange: (frame: number) => void
}

const CinematographyPreview: React.FC<CameraPreviewProps> = ({
  cameraPath,
  currentFrame,
  onFrameChange
}) => {
  const cameraRef = useRef<THREE.PerspectiveCamera>(null!)
  const [isPlaying, setIsPlaying] = useState(false)
  const [previewQuality, setPreviewQuality] = useState<'low' | 'medium' | 'high'>('medium')

  // Camera animation with frame interpolation
  useFrame((state, delta) => {
    if (isPlaying && cameraPath.length > 0) {
      const nextFrame = Math.min(currentFrame + 1, cameraPath.length - 1)
      onFrameChange(nextFrame)

      // Interpolate camera position and rotation
      if (cameraRef.current && cameraPath[currentFrame]) {
        const current = cameraPath[currentFrame]
        const position = new THREE.Vector3(...current.position)
        const rotation = new THREE.Euler(...current.rotation)

        cameraRef.current.position.lerp(position, 0.1)
        cameraRef.current.rotation.x = rotation.x
        cameraRef.current.rotation.y = rotation.y 
        cameraRef.current.rotation.z = rotation.z
        cameraRef.current.fov = current.focal_length * (180 / Math.PI) / 35 * 50  // Approximate conversion
        cameraRef.current.updateProjectionMatrix()
      }
    }
  })

  return (
    <>
      <PerspectiveCamera
        ref={cameraRef}
        makeDefault
        fov={50}
        near={0.1}
        far={1000}
        position={[0, 2, 5]}
      />

      <CameraVisualization 
        cameraPath={cameraPath}
        currentFrame={currentFrame}
      />

      <PreviewControls
        isPlaying={isPlaying}
        onPlayPause={() => setIsPlaying(!isPlaying)}
        quality={previewQuality}
        onQualityChange={setPreviewQuality}
        frame={currentFrame}
        totalFrames={cameraPath.length}
        onFrameSeek={onFrameChange}
      />
    </>
  )
}

const CameraVisualization: React.FC<{
  cameraPath: any[]
  currentFrame: number
}> = ({ cameraPath, currentFrame }) => {
  const pathRef = useRef<THREE.Line>(null!)

  useEffect(() => {
    if (cameraPath.length > 0 && pathRef.current) {
      // Generate camera path visualization
      const points = cameraPath.map(keyframe => 
        new THREE.Vector3(...keyframe.position)
      )

      const geometry = new THREE.BufferGeometry().setFromPoints(points)
      pathRef.current.geometry = geometry
    }
  }, [cameraPath])

  return (
    <group>
      {/* Camera path visualization */}
      <line ref={pathRef}>
        <bufferGeometry />
        <lineBasicMaterial color="#00ff00" linewidth={2} />
      </line>

      {/* Current camera position indicator */}
      {cameraPath[currentFrame] && (
        <mesh position={new THREE.Vector3(...cameraPath[currentFrame].position)}>
          <coneGeometry args={[0.2, 0.5, 4]} />
          <meshBasicMaterial color="#ff0000" />
        </mesh>
      )}

      {/* Frame rulers and guides */}
      <RuleOfThirdsGrid />
    </group>
  )
}

const RuleOfThirdsGrid: React.FC = () => {
```

#### Framework Selection: React Three Fiber + Three.js

**Implementation Approach**

React Three Fiber renders components outside React with no overhead, outperforming Three.js at scale via React's scheduling. It converts JSX to Three.js objects and provides instant access to new Three.js features without library dependencies.

```typescript
// Real-time Preview System with React Three Fiber
import React, { useRef, useState, useEffect, Suspense } from 'react'
import { Canvas, useFrame, useThree } from '@react-three/fiber'
import { OrbitControls, PerspectiveCamera, Environment } from '@react-three/drei'
import * as THREE from 'three'

interface CameraPreviewProps {
  cameraPath: Array<{
    position: [number, number, number]
    rotation: [number, number, number] 
    focal_length: number
    frame: number
  }>
  currentFrame: number
  onFrameChange: (frame: number) => void
}

const CinematographyPreview: React.FC<CameraPreviewProps> = ({
  cameraPath,
  currentFrame,
  onFrameChange
}) => {
  const cameraRef = useRef<THREE.PerspectiveCamera>(null!)
  const [isPlaying, setIsPlaying] = useState(false)
  const [previewQuality, setPreviewQuality] = useState<'low' | 'medium' | 'high'>('medium')

  // Camera animation with frame interpolation
  useFrame((state, delta) => {
    if (isPlaying && cameraPath.length > 0) {
      const nextFrame = Math.min(currentFrame + 1, cameraPath.length - 1)
      onFrameChange(nextFrame)

      // Interpolate camera position and rotation
      if (cameraRef.current && cameraPath[currentFrame]) {
        const current = cameraPath[currentFrame]
        const position = new THREE.Vector3(...current.position)
        const rotation = new THREE.Euler(...current.rotation)

        cameraRef.current.position.lerp(position, 0.1)
        cameraRef.current.rotation.x = rotation.x
        cameraRef.current.rotation.y = rotation.y 
        cameraRef.current.rotation.z = rotation.z
        cameraRef.current.fov = current.focal_length * (180 / Math.PI) / 35 * 50  // Approximate conversion
        cameraRef.current.updateProjectionMatrix()
      }
    }
  })

  return (
    <>
      <PerspectiveCamera
        ref={cameraRef}
        makeDefault
        fov={50}
        near={0.1}
        far={1000}
        position={[0, 2, 5]}
      />

      <CameraVisualization 
        cameraPath={cameraPath}
        currentFrame={currentFrame}
      />

      <PreviewControls
        isPlaying={isPlaying}
        onPlayPause={() => setIsPlaying(!isPlaying)}
        quality={previewQuality}
        onQualityChange={setPreviewQuality}
        frame={currentFrame}
        totalFrames={cameraPath.length}
        onFrameSeek={onFrameChange}
      />
    </>
  )
}

const CameraVisualization: React.FC<{
  cameraPath: any[]
  currentFrame: number
}> = ({ cameraPath, currentFrame }) => {
  const pathRef = useRef<THREE.Line>(null!)

  useEffect(() => {
    if (cameraPath.length > 0 && pathRef.current) {
      // Generate camera path visualization
      const points = cameraPath.map(keyframe => 
        new THREE.Vector3(...keyframe.position)
      )

      const geometry = new THREE.BufferGeometry().setFromPoints(points)
      pathRef.current.geometry = geometry
    }
  }, [cameraPath])

  return (
    <group>
      {/* Camera path visualization */}
      <line ref={pathRef}>
        <bufferGeometry />
        <lineBasicMaterial color="#00ff00" linewidth={2} />
      </line>

      {/* Current camera position indicator */}
      {cameraPath[currentFrame] && (
        <mesh position={new THREE.Vector3(...cameraPath[currentFrame].position)}>
          <coneGeometry args={[0.2, 0.5, 4]} />
          <meshBasicMaterial color="#ff0000" />
        </mesh>
      )}

      {/* Frame rulers and guides */}
      <RuleOfThirdsGrid />
    </group>
  )
}

const RuleOfThirdsGrid: React.FC = () => {
  return (
    <group>
      {/* Horizontal lines */}
      {[-1/3, 1/3].map((y, index) => (
        <line key={`h-${index}`}>
          <bufferGeometry>
            <bufferAttribute
              attach="attributes-position"
              count={2}
              array={new Float32Array([-20, y * 10, 0, 20, y * 10, 0])}
              itemSize={3}
            />
          </bufferGeometry>
          <lineBasicMaterial color="#ffffff" opacity={0.3} transparent />
        </line>
      ))}

      {/* Vertical lines */} 
      {[-1/3, 1/3].map((x, index) => (
        <line key={`v-${index}`}>
          <bufferGeometry>
            <bufferAttribute
              attach="attributes-position"
              count={2}
              array={new Float32Array([x * 10, -20, 0, x * 10, 20, 0])}
              itemSize={3}
            />
          </bufferGeometry>
          <lineBasicMaterial color="#ffffff" opacity={0.3} transparent />
        </line>
      ))}
    </group>
  )
}

const PreviewControls: React.FC<{
  isPlaying: boolean
  onPlayPause: () => void
  quality: 'low' | 'medium' | 'high'
  onQualityChange: (quality: 'low' | 'medium' | 'high') => void
  frame: number
  totalFrames: number
  onFrameSeek: (frame: number) => void
}> = ({ isPlaying, onPlayPause, quality, onQualityChange, frame, totalFrames, onFrameSeek }) => {
  return (
    <div style={{
      position: 'absolute',
      bottom: '20px',
      left: '20px',
      background: 'rgba(0,0,0,0.7)',
      padding: '10px',
      borderRadius: '5px',
      color: 'white'
    }}>
      <button onClick={onPlayPause}>
        {isPlaying ? '⏸️ Pause' : '▶️ Play'}
      </button>

      <input
        type="range"
        min={0}
        max={totalFrames - 1}
        value={frame}
        onChange={(e) => onFrameSeek(parseInt(e.target.value))}
        style={{ margin: '0 10px' }}
      />

      <select 
        value={quality} 
        onChange={(e) => onQualityChange(e.target.value as any)}
      >
        <option value="low">Low Quality</option>
        <option value="medium">Medium Quality</option>
        <option value="high">High Quality</option>
      </select>

      <span>Frame: {frame + 1}/{totalFrames}</span>
    </div>
  )
}

// Main preview application
const CinematographyApp: React.FC = () => {
  const [cameraPath, setCameraPath] = useState<any[]>([])
  const [currentFrame, setCurrentFrame] = useState(0)

  useEffect(() => {
    // Load camera path from Blender export
    fetch('/api/camera-path')
      .then(res => res.json())
      .then(data => setCameraPath(data.keyframes))
  }, [])

  return (
    <div style={{ width: '100vw', height: '100vh' }}>
      <Canvas 
        shadows
        dpr={[1, 2]} 
        camera={{ position: [0, 2, 5], fov: 50 }}
        gl={{ 
          antialias: true,
          alpha: false,
          powerPreference: 'high-performance'
        }}
      >
        <Suspense fallback={null}>
          <Environment preset="studio" />
          <OrbitControls enableDamping />

          <CinematographyPreview
            cameraPath={cameraPath}
            currentFrame={currentFrame}
            onFrameChange={setCurrentFrame}
          />

          {/* Scene content */}
          <SceneVisualization />
        </Suspense>
      </Canvas>
    </div>
  )
}

// Placeholder component
const SceneVisualization: React.FC = () => {
  return (
    <mesh>
      <boxGeometry args={[1, 1, 1]} />
      <meshStandardMaterial color="orange" />
    </mesh>
  )
}

export default CinematographyApp
```

**Pros**
- Exposes the full Three.js API as React components.
- Provides real-time interaction and visual feedback.
- Supports extensive post-processing effects.
- Enables web-based deployment for remote collaboration.

**Cons**
- WebGL text rendering is complex; HTML overlays are often preferred for custom fonts.
- Constrained by browser and WebGL limitations.
- Unsuitable for final production rendering.

```python
    async def process_director_request(self, shot_list: Dict):
        """Main coordination pipeline"""

        # Scene Setup
        scene_objects = await self.scene_manager.setup_scene(
            shot_list['environment']
        )

        # Camera Planning
        camera_plan = await self.cinematography_agent.plan_shots(
            shot_list['shots'],
            scene_objects
        )

        # Lighting Design  
        lighting_plan = await self.lighting_agent.design_lighting(
            shot_list['mood'],
            camera_plan,
            scene_objects
        )

        # Real-time Preview
        preview_data = await self.preview_server.generate_preview(
            camera_plan,
            lighting_plan
        )

        # Director Feedback Loop
        feedback = await self.get_director_feedback(preview_data)

        if feedback['approved']:
            # Final High-Quality Render
            final_renders = await self.render_final_shots(
                camera_plan,
                lighting_plan
            )

        else:
            # Iterate based on feedback
```
