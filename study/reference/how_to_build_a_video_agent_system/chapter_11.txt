### Cinematography Agent Implementation

#### Framework: Blender Python API + CLIP

The API enables full programmatic access to Blender elements like scenes, meshes, and cameras. CLIP maps images and text to shared embeddings for zero-shot classification, trained on 400 million pairs.

```python
# Core Cinematography Agent Architecture
import bpy
import bmesh
import mathutils
from mathutils import Vector, Euler
import numpy as np
import clip
import torch

class CinematographyAgent:
    def __init__(self):
        self.device = "cuda" if torch.cuda.is_available() else "cpu"
        self.clip_model, self.clip_preprocess = clip.load('ViT-B/32', self.device)

    def setup_camera_rig(self, scene_bounds: dict, shot_type: str):
        """Generate camera with precise parametric control."""
        camera_data = bpy.data.cameras.new(name="CineCamera")
        camera_object = bpy.data.objects.new("CineCamera", camera_data)
        bpy.context.collection.objects.link(camera_object)

        if shot_type == "wide_master":
            camera_data.lens = 35  # mm
            camera_data.clip_end = 1000
        elif shot_type == "close_up":
            camera_data.lens = 85  # mm
            camera_data.dof.use_dof = True
            camera_data.dof.aperture_fstop = 2.8

        return camera_object

    def generate_camera_path(self, camera_obj, movement_type: str, duration: int):
        """Generate smooth trajectories using spline interpolation."""
        if movement_type == "handheld":
            for frame in range(1, duration + 1):
                bpy.context.scene.frame_set(frame)
                primary_offset = self._perlin_noise(frame * 0.3) * 0.02
                secondary_offset = self._perlin_noise(frame * 0.8) * 0.015
                breathing_offset = self._perlin_noise(frame * 0.2) * 0.01

                total_offset = Vector((
                    primary_offset + secondary_offset,
                    breathing_offset,
                    secondary_offset * 0.5
                ))

                camera_obj.location += total_offset
                camera_obj.keyframe_insert(data_path="location")

        elif movement_type == "orbital":
            center = Vector((0, 0, 0))
            for frame in range(1, duration + 1):
                angle = (frame / duration) * (270 * np.pi / 180)
                radius = 14 + 2 * np.sin(angle * 5)  # 12-16 feet variance
                z = 5 + 3 * np.sin(angle * 2)  # Height variation

                camera_obj.location = Vector((
                    center.x + radius * np.cos(angle),
                    center.y + radius * np.sin(angle),
                    z
                ))
                camera_obj.keyframe_insert(data_path="location")

    def analyze_composition(self, rendered_image_path: str):
        """Use CLIP for composition analysis."""
        from PIL import Image

        image = Image.open(rendered_image_path)
        image_input = self.clip_preprocess(image).unsqueeze(0).to(self.device)

        cinematography_concepts = [
            "rule of thirds composition",
            "symmetrical framing",
            "leading lines composition",
            "depth of field photography",
            "low angle dramatic shot",
            "high angle overhead shot",
            "close up portrait",
            "wide establishing shot"
        ]

        text_inputs = torch.cat([
            clip.tokenize(f"a photo with {concept}")
            for concept in cinematography_concepts
        ]).to(self.device)

        with torch.no_grad():
            image_features = self.clip_model.encode_image(image_input)
            text_features = self.clip_model.encode_text(text_inputs)

            image_features /= image_features.norm(dim=-1, keepdim=True)
            text_features /= text_features.norm(dim=-1, keepdim=True)
            similarity = (100 * image_features @ text_features.T).softmax(dim=-1)

        return {concept: float(sim) for concept, sim in
                zip(cinematography_concepts, similarity[0])}

    def _perlin_noise(self, t: float) -> float:
        """Simplified Perlin noise for motion simulation."""
        # Implementation details omitted for brevity; use noise library in production.
        return np.sin(t) * 0.5 + 0.5
```

- Extensive documentation and hardware flexibility.

**Cons**:
- Requires Blender environment execution.
- CLIP may need fine-tuning for domain-specific accuracy.
- CPU fallback limits performance without GPU.

### Lighting Agent Implementation

#### Framework: NVIDIA OptiX + Blender Cycles

OptiX provides GPU ray-casting for efficient complex scene processing.

```python
# Advanced Lighting Agent with OptiX Integration
import bpy
import numpy as np
from typing import Dict, List, Tuple
# Note: OptiX bindings require CUDA setup; use bpy for Cycles integration.

class LightingAgent:
    def __init__(self):
        self.setup_cycles()

    def setup_cycles(self):
        """Initialize Cycles with OptiX if available."""
        scene = bpy.context.scene
        scene.render.engine = 'CYCLES'

        cycles_prefs = bpy.context.preferences.addons['cycles'].preferences
        if any(device.type == 'OPTIX' for device in cycles_prefs.devices):
            scene.cycles.device = 'GPU'
            scene.cycles.feature_set = 'EXPERIMENTAL'
            scene.cycles.use_optix_ai_denoising = True

    def setup_three_point_lighting(self, scene_context: dict):
        """Automated 3-point lighting with IES profiles."""
        # Key Light
        key_light = bpy.data.lights.new(name="KeyLight", type='AREA')
        key_obj = bpy.data.objects.new("KeyLight", key_light)
        bpy.context.collection.objects.link(key_obj)
        key_obj.location = (10, -5, 5)
        key_light.energy = 2200
        key_light.color = (1, 0.95, 0.9)  # 5400K approximation
        key_light.size = 8  # Softbox equivalent

        # Fill Light (Fireplace)
        fill_light = bpy.data.lights.new(name="FillLight", type='POINT')
        fill_obj = bpy.data.objects.new("FillLight", fill_light)
        bpy.context.collection.objects.link(fill_obj)
        fill_obj.location = (-5, 0, 1)
        fill_light.energy = 500  # Mid-range
        fill_light.color = (0.9, 0.4, 0.2)  # 1950K warm

        # Rim Lights (Chandelier)
        rim_lights = []
        for i in range(8):
            rim_light = bpy.data.lights.new(name=f"RimLight_{i}", type='POINT')
            rim_obj = bpy.data.objects.new(f"RimLight_{i}", rim_light)
            bpy.context.collection.objects.link(rim_obj)
            angle = (i / 8) * 2 * np.pi
            rim_obj.location = (3 * np.cos(angle), 3 * np.sin(angle), 8)
            rim_light.energy = 150
            rim_light.color = (1, 0.85, 0.7)  # 2800K
            rim_lights.append(rim_obj)

        return {'key': key_obj, 'fill': fill_obj, 'rims': rim_lights}

    def configure_cycles_path_tracing(self):
        """Advanced path tracing setup."""
        scene = bpy.context.scene
        scene.cycles.max_bounces = 12
        scene.cycles.diffuse_bounces = 8
        scene.cycles.glossy_bounces = 6
        scene.cycles.transmission_bounces = 4
        scene.cycles.transparent_max_bounces = 8
        scene.cycles.samples = 512
        scene.cycles.use_adaptive_sampling = True
        scene.cycles.adaptive_threshold = 0.01

    def simulate_physically_accurate_lighting(self, light_setup: dict):
        """Physically based transport with HDRI."""
        world = bpy.context.scene.world
        world.use_nodes = True
        nodes = world.node_tree.nodes
        links = world.node_tree.links

        # HDRI setup
        env_tex = nodes.new(type='ShaderNodeTexEnvironment')
        env_tex.image = bpy.data.images.load("victorian_interior.hdr")  # Path to HDRI

        output = nodes.get('World Output')
        background = nodes.get('Background')
        links.new(background.outputs['Color'], output.inputs['Color'])
        links.new(env_tex.outputs['Color'], background.inputs['Color'])

        # Light sampling
        for light_name, light_obj in light_setup.items():
            if light_obj.data.cycles:
                light_obj.data.cycles.use_multiple_importance_sampling = True
                light_obj.data.cycles.samples = 16 if light_name == 'key' else 8

    def render_with_optix_acceleration(self, output_path: str):
        """Production render."""
        scene = bpy.context.scene
        scene.render.resolution_x = 3840
        scene.render.resolution_y = 2160
        scene.render.image_settings.file_format = 'OPEN_EXR'
        scene.render.image_settings.color_depth = '32'
        scene.render.image_settings.exr_codec = 'DWAA'

        # Render passes
        view_layer = scene.view_layers["ViewLayer"]
        view_layer.use_pass_diffuse_direct = True
        view_layer.use_pass_diffuse_indirect = True
        view_layer.use_pass_glossy_direct = True
        view_layer.use_pass_glossy_indirect = True
        view_layer.use_pass_shadow = True

        scene.render.filepath = output_path
        bpy.ops.render.render(write_still=True)

        return output_path
```

```python
# Scene Management with BlenderProc Integration
import bproc
import numpy as np
from typing import Dict, List
import json

class SceneManager:
    def __init__(self):
        bproc.init()
        self.scene_config = {}

    def setup_victorian_mansion_scene(self, config_path: str):
        """Load and configure environment."""
        with open(config_path, 'r') as f:
            self.scene_config = json.load(f)

        mansion_path = self.scene_config['mansion_model_path']
        objs = bproc.loader.load_3d_front(
            json_path=mansion_path,
            future_model_path=self.scene_config['furniture_path']
        )

        self.configure_pbr_materials(objs)
        self.setup_procedural_lighting()

        return objs

    def configure_pbr_materials(self, objects: List):
        """PBR material setup."""
        for obj in [o for o in objects if o.get_mesh()]:
            mat = bproc.material.create("PBRMaterial")
            obj.set_material(0, mat)

            name_lower = obj.get_name().lower()
            if "wood" in name_lower:
                mat.set_principled_shader_value("Base Color", [0.4, 0.2, 0.1, 1])
                mat.set_principled_shader_value("Roughness", 0.7)
                mat.set_principled_shader_value("Specular", 0.3)
            elif "fabric" in name_lower:
                mat.set_principled_shader_value("Base Color", [0.3, 0.1, 0.1, 1])
                mat.set_principled_shader_value("Roughness", 0.9)
                mat.set_principled_shader_value("Specular", 0.1)
            elif "metal" in name_lower:
                mat.set_principled_shader_value("Base Color", [0.7, 0.7, 0.7, 1])
                mat.set_principled_shader_value("Metallic", 0.9)
                mat.set_principled_shader_value("Roughness", 0.2)

    def setup_procedural_lighting(self):
        """Domain-randomized lighting."""
        window_locations = self.scene_config.get('window_positions', [])
        for pos in window_locations:
            light = bproc.types.Light()
            light.set_type("AREA")
            light.set_location(pos)
            light.set_energy(np.random.uniform(800, 1500))
            light.set_color(self._color_temp_to_rgb(np.random.uniform(5200, 5800)))

        self._add_practical_lights()

    def _add_practical_lights(self):
        """Motivated sources."""
        if 'chandelier_position' in self.scene_config:
            chand_pos = self.scene_config['chandelier_position']
            for i in range(8):
                angle = (i / 8) * 2 * np.pi
                offset_x = 0.5 * np.cos(angle)
                offset_y = 0.5 * np.sin(angle)

                light = bproc.types.Light()
                light.set_type("POINT")
                light.set_location([
                    chand_pos[0] + offset_x,
                    chand_pos[1] + offset_y,
                    chand_pos[2]
                ])
                light.set_energy(np.random.uniform(140, 160))
                light.set_color(self._color_temp_to_rgb(2800))

        if 'fireplace_position' in self.scene_config:
            light = bproc.types.Light()
            light.set_type("POINT")
            light.set_location(self.scene_config['fireplace_position'])

            for frame in range(1, 250):
                energy = np.random.uniform(400, 700)
                light.set_energy(energy, frame)
                temp = np.random.uniform(1900, 2200)
                light.set_color(self._color_temp_to_rgb(temp), frame)

    def generate_camera_sampling_constraints(self) -> Dict:
        """Scene constraints."""
        room_bounds = self.scene_config.get('room_dimensions', {
            'min_x': -10, 'max_x': 10,
            'min_y': -8, 'max_y': 8,
            'min_z': 0.5, 'max_z': 7
        })

        furniture_bounds = [
            obj.get_bound_box() for obj in bproc.object.get_all_mesh_objects()
            if 'furniture' in obj.get_name().lower()
        ]

        return {
            'room_bounds': room_bounds,
            'obstacles': furniture_bounds,
            'min_subject_distance': 2,  # feet
            'max_subject_distance': 20,
            'preferred_height_range': (5, 7),  # feet
            'excluded_zones': self.scene_config.get('no_camera_zones', [])
        }

    def _color_temp_to_rgb(self, temp_k: float) -> List[float]:
        """Blackbody approximation."""
        if temp_k < 3000:
            return [1.0, 0.7, 0.4]  # Warm
        elif temp_k < 4000:
            return [1.0, 0.9, 0.7]
        elif temp_k < 5500:
            return [1.0, 1.0, 1.0]
        else:
            return [0.95, 1.0, 1.0]  # Cool
```

```typescript
  rotation: [number, number, number];

  cameraPath: CameraKeyframe[];
  currentFrame: number;
  onFrameChange: (frame: number) => void;

const CinematographyPreview: React.FC<CameraPreviewProps> = ({
  cameraPath,
  currentFrame,
  onFrameChange,
}) => {
  const cameraRef = useRef<THREE.PerspectiveCamera>(null!);
  const [isPlaying, setIsPlaying] = useState(false);
  const [previewQuality, setPreviewQuality] = useState<'low' | 'medium' | 'high'>('medium');

  useFrame((state, delta) => {
    if (isPlaying && cameraPath.length > 0) {
      const nextFrame = Math.min(currentFrame + 1, cameraPath.length - 1);
      onFrameChange(nextFrame);

      if (cameraRef.current && cameraPath[currentFrame]) {
        const current = cameraPath[currentFrame];
        const position = new THREE.Vector3(...current.position);
        const rotation = new THREE.Euler(...current.rotation);

        cameraRef.current.position.lerp(position, 0.1);
        cameraRef.current.rotation.set(rotation.x, rotation.y, rotation.z);
        cameraRef.current.fov = current.focal_length * (Math.PI / 180) * 0.8;  // Approximate
        cameraRef.current.updateProjectionMatrix();
      }
    }
  });

  return (
    <>
      <PerspectiveCamera
        ref={cameraRef}
        makeDefault
        fov={50}
        near={0.1}
        far={1000}
        position={[0, 2, 5]}
      />
      <CameraVisualization cameraPath={cameraPath} currentFrame={currentFrame} />
      <PreviewControls
        isPlaying={isPlaying}
        onPlayPause={() => setIsPlaying(!isPlaying)}
        quality={previewQuality}
        onQualityChange={setPreviewQuality}
        frame={currentFrame}
        totalFrames={cameraPath.length}
        onFrameSeek={onFrameChange}
      />
    </>
  );
};

const CameraVisualization: React.FC<{
  cameraPath: CameraKeyframe[];
  currentFrame: number;

      pathRef.current.geometry = geometry;

  return (
    <group>
      <line ref={pathRef}>
        <bufferGeometry />
        <lineBasicMaterial color="#00ff00" linewidth={2} />
      </line>
      {cameraPath[currentFrame] && (
        <mesh position={cameraPath[currentFrame].position}>
          <coneGeometry args={[0.1, 0.3, 4]} />
          <meshBasicMaterial color="#ff0000" />
        </mesh>
      )}
      <RuleOfThirdsGrid />
    </group>
  );
};
```
