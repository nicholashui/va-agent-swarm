    async def initialize_session(self):
        """Initialize HTTP session with optimized configuration."""
        if self.session is None:
            connector = aiohttp.TCPConnector(
                limit=100,
                limit_per_host=30,
                ttl_dns_cache=300,
                use_dns_cache=True,
                keepalive_timeout=30
            )
            
            timeout = aiohttp.ClientTimeout(total=30, connect=10)
            
            headers = {
                'User-Agent': (
                    'Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36 '
                    '(KHTML, like Gecko) Chrome/120.0.0.0 Safari/537.36'

                'Connection': 'keep-alive',
                'Upgrade-Insecure-Requests': '1'
            }
            
            self.session = aiohttp.ClientSession(
                connector=connector,
                timeout=timeout,
                headers=headers
            )
    
    async def scrape_box_office_mojo(self, session: aiohttp.ClientSession) -> List[Dict[str, Any]]:
        """
        Scrape daily box office data from Box Office Mojo.
        
        Returns structured data with error handling and data validation.
        """
        url = self.sources["box_office_mojo"]["url"]
        
        try:
            async with session.get(url) as response:
                if response.status != 200:
                    print(f"HTTP {response.status} for Box Office Mojo")
                    return []
                
                html = await response.text()
                soup = BeautifulSoup(html, 'html.parser')
                
                table = soup.find('table', class_='a-bordered') or \
                       soup.find('table', class_='imdb-scroll-table')
                
                if not table:
                    print("No table found on Box Office Mojo")
                    return []
                
                rows = table.find_all('tr')[1:21]  # Top 20 films
                data = []
                
                for row in rows:
                    cols = row.find_all(['td', 'th'])
                    if len(cols) < 8:
                        continue
                    
                    try:
                        film_data = {
                            'rank': self._safe_int_extract(cols[0].text.strip()),
                            'title': self._extract_title(cols[1]),
                            'studio': cols[2].text.strip() if len(cols) > 2 else '',
                            'daily_gross': self._parse_currency(cols[3].text.strip()),
                            'change': self._parse_percentage(cols[4].text.strip()),
                            'theaters': self._safe_int_extract(
                                cols[5].text.strip().replace(',', '')
                            ),
                            'per_theater': self._parse_currency(cols[6].text.strip()),
                            'total_gross': self._parse_currency(cols[7].text.strip()),
                            'days_in_release': self._safe_int_extract(cols[8].text.strip()) if len(cols) > 8 else 1,
                            'timestamp': datetime.now().isoformat(),
                            'source': 'box_office_mojo',
                            'data_quality_score': self._calculate_data_quality(row)
                        }
                        
                        # Validate essential fields
                        if film_data['title'] and film_data['daily_gross'] is not None:
                            data.append(film_data)
                            
                    except (ValueError, IndexError, AttributeError) as e:
                        print(f"Error parsing row: {e}")
                        continue
                
                print(f"Scraped {len(data)} records from Box Office Mojo")
                return data
                
        except Exception as e:
            print(f"Error scraping Box Office Mojo: {e}")
            return []

                        title = title_link.text.strip() if title_link else cols[1].text.strip()

                            'title': title,

    async def get_theater_chain_data(self) -> List[Dict[str, Any]]:
        """
        Aggregate real-time data from major theater chain APIs.
        
        Provides venue-level granularity and demographic insights.
        """
        theater_data = []
        chain_configs = self.sources["theater_chains"]["endpoints"]
        
        async with aiohttp.ClientSession() as session:
            tasks = []
            
            for chain_name, api_url in chain_configs.items():
                api_key = os.getenv(f"{chain_name.upper()}_API_KEY")
                if not api_key:
                    continue
                
                task = self._fetch_chain_data(session, chain_name, api_url, api_key)
                tasks.append(task)
            
            # Execute all chain API calls concurrently
            results = await asyncio.gather(*tasks, return_exceptions=True)
            
            # Process successful results
            for result in results:
                if isinstance(result, list):
                    theater_data.extend(result)
        
        print(f"Aggregated {len(theater_data)} theater records")
        return theater_data
    
    async def _fetch_chain_data(self, session: aiohttp.ClientSession, 
                               chain_name: str, api_url: str, api_key: str) -> List[Dict[str, Any]]:
        """Fetch data from individual theater chain API."""
        try:
            headers = {
                'Authorization': f'Bearer {api_key}',
                'Content-Type': 'application/json'
            }
            
            async with session.get(api_url, headers=headers) as response:
                if response.status == 200:
                    chain_data = await response.json()
                    processed_data = []
                    
                    for theater in chain_data.get('theaters', []):
                        theater_record = {
                            'chain': chain_name.upper(),
                            'theater_id': theater.get('id'),
                            'theater_name': theater.get('name'),
                            'location': theater.get('address'),
                            'city': theater.get('city'),
                            'state': theater.get('state'),
                            'zip_code': theater.get('zip'),
                            'current_sales': theater.get('current_sales', {}),
                            'showtimes': theater.get('showtimes', []),
                            'capacity_utilization': theater.get('occupancy_rate'),
                            'demographics': theater.get('demographics', {}),
                            'timestamp': datetime.now().isoformat(),
                            'source': f'theater_{chain_name}'
                        }
                        
                        processed_data.append(theater_record)
                    
                    return processed_data
                else:
                    print(f"{chain_name} API returned status {response.status}")
                    return []
                    
        except Exception as e:
            print(f"Error fetching {chain_name} data: {e}")
            return []

            return pd.DataFrame()
        
        # Create unified DataFrame
        df = pd.DataFrame(all_data)
        
        if df.empty:
            return df
        
        # Advanced data processing and validation
        df = await self._process_and_validate_data(df)
        
        # Cache processed data
        self.data_cache['latest'] = df
        
        print(f"Successfully processed {len(df)} total records from {len(set(df['source'].dropna()))} sources")
        return df
    
    async def _process_and_validate_data(self, df: pd.DataFrame) -> pd.DataFrame:
        """
        Advanced data processing with deduplication, validation, and quality scoring.
        
        Implements sophisticated data fusion algorithms to resolve conflicts
        between multiple sources reporting the same film.
        """
        if df.empty:
            return df
        
        # Standardize column names and data types
        df = df.rename(columns={
            'daily_gross': 'daily_gross',
            'theaters': 'theater_count', 
            'per_theater': 'per_theater_average',
            'total_gross': 'cumulative_gross'
        })
        
        # Create timestamp window for deduplication (5-minute windows)
        df['timestamp'] = pd.to_datetime(df['timestamp'])
        df['time_window'] = df['timestamp'].dt.floor('5min')
        
        # Source priority for conflict resolution
        source_priority = {
            'comscore': 1,           # Professional service - highest priority
            'theater_amc': 2,        # Direct theater data
            'theater_regal': 2,
            'theater_cinemark': 2,
            'the_numbers': 3,        # Reliable industry source
            'box_office_mojo': 4,    # Popular but sometimes delayed
            'deadline': 5            # News source - lowest priority
        }
        
        df['source_priority'] = df['source'].map(source_priority).fillna(10)
        
        # Deduplicate by film title and time window, keeping highest priority source
        df_clean = df.sort_values(
            ['title', 'time_window', 'source_priority', 'data_quality_score']
        ).drop_duplicates(
            subset=['title', 'time_window'], 
            keep='first'
        )
        
        # Data quality validation and outlier detection
        numeric_columns = ['daily_gross', 'theater_count', 'per_theater_average']
        for col in numeric_columns:
            if col in df_clean.columns:
                # Remove negative values and extreme outliers using IQR method
                Q1 = df_clean[col].quantile(0.25)
                Q3 = df_clean[col].quantile(0.75)
                IQR = Q3 - Q1
                
                if pd.notna(IQR) and IQR > 0:
                    lower_bound = Q1 - 3 * IQR
                    upper_bound = Q3 + 3 * IQR
                    
                    # Filter outliers but flag for review
                    df_clean[f'{col}_outlier'] = (
                        (df_clean[col] < lower_bound) | 
                        (df_clean[col] > upper_bound) |
                        (df_clean[col] < 0)
                    )
        
        # Validate per-theater calculations
        if all(col in df_clean.columns for col in ['daily_gross', 'theater_count', 'per_theater_average']):
            df_clean['calculated_per_theater'] = df_clean['daily_gross'] / df_clean['theater_count']
            df_clean['per_theater_consistent'] = (
                abs(df_clean['per_theater_average'] - df_clean['calculated_per_theater']) 
                / df_clean['calculated_per_theater'] < 0.1  # 10% tolerance
            )
        
        # Calculate overall data quality score
        quality_factors = [
            'source_priority', 
            'per_theater_consistent',
            'data_quality_score'
        ]
        
        df_clean['overall_quality_score'] = df_clean.apply(
            lambda row: self._calculate_overall_quality(row, quality_factors), axis=1
        )
        
        # Final filtering - keep only high-quality records
        df_final = df_clean[
            (df_clean['overall_quality_score'] > 0.7) & 
            (df_clean['title'].str.len() > 0)
        ].copy()
        
        # Add derived metrics
        df_final['market_share_estimate'] = (
            df_final['daily_gross'] / df_final['daily_gross'].sum()
        ) * 100
        
        df_final['performance_category'] = pd.cut(
            df_final['per_theater_average'],
            bins=[0, 5000, 10000, 20000, float('inf')],
            labels=['Poor', 'Average', 'Strong', 'Excellent']
        )
        
        print(f"Data processing complete: {len(df_final)} validated records")
        return df_final
    
    def _safe_int_extract(self, text: str) -> int:
        """Safely extract integers from text with error handling."""
        try:
            cleaned = text.strip().replace(',', '').replace('$', '')
            return int(cleaned) if cleaned.isdigit() else 0
        except (ValueError, AttributeError):
            return 0
    
    def _extract_title(self, element) -> str:
        """Extract and clean film title from HTML element."""
        try:
            if hasattr(element, 'find') and element.find('a'):
                return element.find('a').text.strip()
            return element.text.strip() if hasattr(element, 'text') else str(element).strip()
        except:
            return ""
    
    def _parse_currency(self, text: str) -> float:
        """Parse various currency formats including M/B/K notation."""
        try:
            # Clean the text
            cleaned = str(text).replace('$', '').replace(',', '').strip()
            
            # Handle M/B/K notation
            if 'M' in cleaned.upper():
                return float(cleaned.upper().replace('M', '')) * 1_000_000
            elif 'B' in cleaned.upper():
                return float(cleaned.upper().replace('B', '')) * 1_000_000_000
            elif 'K' in cleaned.upper():
                return float(cleaned.upper().replace('K', '')) * 1_000
            else:
                return float(cleaned) if cleaned else 0.0
                
        except (ValueError, AttributeError):
            return 0.0
    
    def _parse_percentage(self, text: str) -> float:
        """Parse percentage changes with sign detection."""
        try:
            cleaned = str(text).replace('%', '').strip()
            if cleaned.startswith('-'):
                sign = -1
                cleaned = cleaned[1:]
            elif cleaned.startswith('+'):
                sign = 1
                cleaned = cleaned[1:]
            else:
                sign = 1
            
            value = float(cleaned) if cleaned else 0.0
            return value * sign
            
        except (ValueError, AttributeError):
            return 0.0
    
    def _calculate_data_quality(self, element) -> float:
        """Calculate data quality score based on HTML element completeness."""
        try:
            # Basic completeness check
            text_content = element.get_text(strip=True)
            missing_elements = text_content.count('-') + text_content.count('N/A')
            total_elements = len(element.find_all(['td', 'th']))
            
            completeness = 1.0 - (missing_elements / max(total_elements, 1))
            return max(0.1, completeness)  # Minimum quality score
        except:
            return 0.5
    
    def _calculate_overall_quality(self, row, quality_factors: List[str]) -> float:
        """Calculate composite quality score from multiple factors."""
        score = 0.0
        weight_total = 0.0
        
        # Source priority (inverted - lower number = higher priority)
        if 'source_priority' in quality_factors and pd.notna(row['source_priority']):
            source_weight = 1.0 / row['source_priority']
            score += source_weight * 0.4  # 40% weight
            weight_total += 0.4
        
        # Data consistency checks
        if 'per_theater_consistent' in quality_factors:
            consistency_weight = 1.0 if row.get('per_theater_consistent', False) else 0.5
            score += consistency_weight * 0.3  # 30% weight
            weight_total += 0.3
        
        # Original data quality score
        if 'data_quality_score' in quality_factors and pd.notna(row['data_quality_score']):
            score += row['data_quality_score'] * 0.3  # 30% weight
            weight_total += 0.3
        
        return score / max(weight_total, 0.1)
    
    async def close(self):
        """Clean up resources."""
        if self.session:
            await self.session.close()

                revenue_today, tickets_sold_today, occupancy_rate_today_avg,
                avg_ticket_price, concession_revenue,
                demographic_breakdown,

                LAG(revenue_today) OVER (PARTITION BY theater_id ORDER BY timestamp) as prev_revenue,

        total_revenue = data['revenue_today'].sum()
        avg_occupancy = data['occupancy_rate_today_avg'].mean()
        avg_ticket_price = data['avg_ticket_price'].mean()
        
        # Calculate revenue efficiency
        theaters_reporting = len(data['theater_id'].nunique())
        avg_revenue_per_theater = total_revenue / theaters_reporting if theaters_reporting > 0 else 0
        
        # Concession revenue analysis
        total_concessions = data['concession_revenue'].sum()
        concession_ratio = total_concessions / total_revenue if total_revenue > 0 else 0
        
        # Demographic aggregation
        avg_demographics = data['demographic_breakdown'].apply(
            lambda x: pd.Series(x) if isinstance(x, dict) else pd.Series()
        ).mean().to_dict()
        
        # Performance momentum (hourly growth rate)
        if 'prev_revenue' in data.columns:
            data['revenue_growth'] = (
                (data['revenue_today'] - data['prev_revenue']) / 
                data['prev_revenue'].replace(0, 1)
            )
            avg_growth_rate = data['revenue_growth'].mean()
        else:
            avg_growth_rate = 0.0
        
        return {
            "total_tickets_sold": int(total_tickets),
            "total_revenue": round(total_revenue, 2),
            "average_occupancy": round(avg_occupancy * 100, 1),  # Percentage
            "average_ticket_price": round(avg_ticket_price, 2),
            "theaters_reporting": theaters_reporting,
            "average_revenue_per_theater": round(avg_revenue_per_theater, 2),
            "total_concession_revenue": round(total_concessions, 2),
            "concession_to_ticket_ratio": round(concession_ratio * 100, 1),
            "average_growth_rate": round(avg_growth_rate * 100, 1),
            "average_demographics": {k: round(v, 3) for k, v in avg_demographics.items()},

    def _classify_overall_performance(self, occupancy: float, growth_rate: float) -> str:

        elif occupancy_score > 60 or growth_rate > 0:
            return "Strong"

            return "Average"

    def _create_temporal_features(self, df: pd.DataFrame) -> pd.DataFrame:
        """Generate temporal features including seasonality and holiday effects."""
        # Release timing features
        df['release_date'] = pd.to_datetime(df['release_date'])
        df['release_month'] = df['release_date'].dt.month
        df['release_day'] = df['release_date'].dt.day
        df['release_day_of_week'] = df['release_date'].dt.dayofweek
        df['is_friday_release'] = (df['release_day_of_week'] == 4).astype(int)
        df['is_thursday_release'] = (df['release_day_of_week'] == 3).astype(int)
        
        # Seasonal indicators
        df['is_summer_blockbuster_season'] = df['release_month'].isin([5, 6, 7, 8]).astype(int)
        df['is_holiday_season'] = df['release_month'].isin([11, 12]).astype(int)
        df['is_q1_slow_season'] = df['release_month'].isin([1, 2]).astype(int)
        
        # Holiday proximity (major US holidays affecting attendance)
        holiday_distances = []
        major_holidays = [
            pd.Timestamp('2023-01-01'),  # New Year's Day
            pd.Timestamp('2023-07-04'),  # Independence Day
            pd.Timestamp('2023-11-23'),  # Thanksgiving
            pd.Timestamp('2023-12-25'),  # Christmas
            pd.Timestamp('2023-05-29'),  # Memorial Day
            pd.Timestamp('2023-09-04'),  # Labor Day
        ]
        
        for date in df['release_date']:
            distances = [abs((date - holiday).days) for holiday in major_holidays]
            min_distance = min(distances)
            holiday_distances.append(min_distance)
        
        df['days_to_nearest_holiday'] = holiday_distances
        df['holiday_proximity_bonus'] = (df['days_to_nearest_holiday'] <= 3).astype(int)
        df['holiday_weekend_potential'] = (df['days_to_nearest_holiday'] <= 7).astype(int)
        
        # Weather impact proxy (historical averages by month/region)
        monthly_weather_scores = {
            1: 0.6, 2: 0.5, 3: 0.7, 4: 0.8, 5: 0.9,
            6: 1.0, 7: 1.0, 8: 0.9, 9: 0.8, 10: 0.7,
            11: 0.6, 12: 0.5
        }
        df['seasonal_weather_score'] = df['release_month'].map(monthly_weather_scores)
        
        # School calendar impact
        df['is_summer_vacation'] = df['release_month'].isin([6, 7, 8]).astype(int)
        df['is_back_to_school'] = df['release_month'].isin([8, 9]).astype(int)
        
        return df
    
    def _create_competition_features(self, df: pd.DataFrame) -> pd.DataFrame:
        """Generate competitive landscape features using market share analysis."""
        # Calculate competition intensity by release date
        df['release_date_group'] = df['release_date'].dt.floor('D')
        
        # Number of competing releases by budget tier
        df['competing_low_budget'] = df.groupby('release_date_group')['production_budget'].transform(
            lambda x: ((x < 50_000_000) & (x > 0)).sum() - 1  # Exclude self
        )
        df['competing_mid_budget'] = df.groupby('release_date_group')['production_budget'].transform(
            lambda x: ((x >= 50_000_000) & (x < 150_000_000)).sum() - 1
        )
        df['competing_high_budget'] = df.groupby('release_date_group')['production_budget'].transform(
            lambda x: (x >= 150_000_000).sum() - 1
        )
        
        # Total competing budget impact
        df['total_competing_budget'] = df.groupby('release_date_group')['production_budget'].transform('sum') - df['production_budget']
        df['budget_competition_ratio'] = df['production_budget'] / (df['total_competing_budget'] + 1)
        
        # Genre competition analysis
        genre_competition = df.groupby(['release_date_group', 'genre_primary']).size().unstack(fill_value=0)
        for genre in genre_competition.columns:
            df[f'competition_{genre.lower()}'] = df.groupby('release_date_group').apply(
                lambda x: genre_competition.loc[x.name, x['genre_primary']] - 1 if x.name in genre_competition.index else 0
            ).values
        
        # Market saturation index (theaters per film)
        df['total_theater_competition'] = df.groupby('release_date_group')['theater_count'].transform('sum') - df['theater_count']
        df['theater_saturation_index'] = df['theater_count'] / (df['total_theater_competition'] + df['theater_count'])
        
        # Similar film competition (same genre + budget tier)
        df['budget_tier'] = pd.cut(df['production_budget'], 
                                 bins=[0, 50e6, 150e6, float('inf')], 
                                 labels=['low', 'mid', 'high'])
        
        similar_competition = df.groupby(['release_date_group', 'genre_primary', 'budget_tier']).size()
        df['similar_film_competition'] = df.apply(
            lambda row: similar_competition.get((row['release_date_group'], row['genre_primary'], row['budget_tier']), 0) - 1,
            axis=1
        )
        
        return df
    
    def _create_marketing_features(self, df: pd.DataFrame) -> pd.DataFrame:
        """Generate marketing effectiveness and reach features."""
        # Budget efficiency ratios
        df['marketing_to_production_ratio'] = df['marketing_budget'] / (df['production_budget'] + 1e6)
        df['total_budget'] = df['production_budget'] + df['marketing_budget']
        df['p_a_ratio'] = df['production_budget'] / (df['marketing_budget'] + 1e6)
        
        # Digital marketing effectiveness
        df['digital_marketing_efficiency'] = df['digital_ad_reach'] / (df['marketing_budget'] + 1e6)
        df['social_media_roi'] = df['social_buzz_intensity'] / (df['digital_ad_reach'] + 1)
        
        # Trailer performance metrics
        df['trailer_engagement_score'] = df['trailer_engagement_rate'] * np.log1p(df['trailer_views_millions'])
        df['trailer_conversion_potential'] = df['trailer_views_millions'] * df['trailer_engagement_rate']
        
        # Multi-channel marketing synergy
        df['marketing_channel_diversity'] = (
            (df['tv_spot_impressions'] > 0).astype(int) +
            (df['digital_ad_reach'] > 0).astype(int) +
            (df['social_buzz_intensity'] > 0).astype(int) +
            (df['partnership_value'] > 0).astype(int)
        )
        
        # Pre-release hype accumulation
        df['cumulative_hype_score'] = (
            df['pre_release_hype_score'] * 
            np.log1p(df['marketing_channel_diversity']) *
            df['marketing_to_production_ratio']
        )
        
        # Timing of marketing spend
        df['marketing_timing_efficiency'] = 1 - abs(df['release_month'] - 6) / 12  # Peak in summer
        
        return df
    
    def train_production_model(self, historical_data: pd.DataFrame) -> Dict[str, Any]:
        """
        Train the complete ensemble model with hyperparameter optimization.
        
        Uses walk-forward validation to prevent temporal data leakage.
        Implements ensemble of XGBoost with feature selection and uncertainty estimation.
        """
        print("Starting model training with advanced ensemble methods...")
        
        # Feature engineering
        df = self.engineer_advanced_features(historical_data)
        
        # Prepare target variable
        y = df['opening_weekend_gross'].fillna(0)
        df = df.drop(columns=['opening_weekend_gross'])
        
        # Handle categorical variables
        categorical_columns = ['genre_primary', 'mpaa_rating', 'budget_tier', 'release_month']
        for col in categorical_columns:
            if col in df.columns:
                le = LabelEncoder()
                df[f'{col}_encoded'] = le.fit_transform(df[col].astype(str))
                self.label_encoders[col] = le
        
        # Select available features
        available_features = [f for f in self.all_features if f in df.columns]
        X = df[available_features].fillna(0)
        
        print(f"Training with {len(available_features)} features and {len(X)} samples")
        
        # Temporal train-test split (80% train, 20% test, chronological)
        df_sorted = df.sort_values('release_date')
        split_idx = int(len(df_sorted) * 0.8)
        
        X_train = X.iloc[:split_idx]
        X_test = X.iloc[split_idx:]
        y_train = y.iloc[:split_idx]
        y_test = y.iloc[split_idx:]
        
        # Scale features
        X_train_scaled = self.scaler.fit_transform(X_train)
        X_test_scaled = self.scaler.transform(X_test)
        
        # Hyperparameter optimization using Optuna
        best_params = self._optimize_hyperparameters(X_train_scaled, y_train)
        
        # Train primary XGBoost model
        self.primary_model = xgb.XGBRegressor(
            **best_params,
            random_state=42,
            n_jobs=-1,
            verbosity=0
        )
        
        # Use time series cross-validation for robust training
        tscv = TimeSeriesSplit(n_splits=5)
        self.primary_model.fit(
            X_train_scaled, y_train,
            eval_set=[(X_test_scaled, y_test)],
            early_stopping_rounds=50,
            verbose=False
        )
        
        # Train ensemble models for uncertainty estimation
        self._train_ensemble_models(X_train_scaled, y_train)
        
        # Evaluate model performance
        performance = self._evaluate_model_performance(X_test_scaled, y_test)
        
        # Extract feature importance
        self.feature_importance = dict(zip(
            available_features, 
            self.primary_model.feature_importances_
        ))
        
        # Model metadata
        metadata = {
            'training_samples': len(X_train),
            'test_samples': len(X_test),
            'features_used': len(available_features),
            'best_params': best_params,
            'performance_metrics': performance,
            'feature_importance': dict(
                sorted(self.feature_importance.items(), key=lambda x: x[1], reverse=True)[:20]
            ),
            'training_timestamp': datetime.now().isoformat()
        }
        
        print(f"Model training completed. R²: {performance['r2_score']:.3f}, MAE: ${performance['mae']:,.0f}")
        return metadata
    
    def _optimize_hyperparameters(self, X: np.ndarray, y: pd.Series) -> Dict[str, Any]:
        """Advanced hyperparameter optimization using Bayesian optimization."""
        
        def objective(trial):
            params = {
                'objective': 'reg:squarederror',
                'n_estimators': trial.suggest_int('n_estimators', 500, 2000),
                'max_depth': trial.suggest_int('max_depth', 3, 10),
                'learning_rate': trial.suggest_float('learning_rate', 0.01, 0.3, log=True),
                'subsample': trial.suggest_float('subsample', 0.6, 1.0),
                'colsample_bytree': trial.suggest_float('colsample_bytree', 0.6, 1.0),
                'reg_alpha': trial.suggest_float('reg_alpha', 0, 10),
                'reg_lambda': trial.suggest_float('reg_lambda', 0, 10),
                'min_child_weight': trial.suggest_int('min_child_weight', 1, 10),
                'gamma': trial.suggest_float('gamma', 0, 5),
                'random_state': 42
            }
            
            # Time series split for validation
            tscv = TimeSeriesSplit(n_splits=3)
            scores = []
            
            for train_idx, val_idx in tscv.split(X):
                X_train_fold, X_val_fold = X[train_idx], X[val_idx]
                y_train_fold, y_val_fold = y.iloc[train_idx], y.iloc[val_idx]
                
                model = xgb.XGBRegressor(**params, verbosity=0)
                model.fit(X_train_fold, y_train_fold, 
                         eval_set=[(X_val_fold, y_val_fold)],
                         early_stopping_rounds=20, verbose=False)
                
                val_pred = model.predict(X_val_fold)
                mae = mean_absolute_error(y_val_fold, val_pred)
                scores.append(mae)
            
            return np.mean(scores)
        
        # Optimize with 100 trials
        study = optuna.create_study(direction='minimize')
        study.optimize(objective, n_trials=100, show_progress_bar=True)
        
        print(f"Best hyperparameters: {study.best_params}")
        return study.best_params
    
    def predict_opening_weekend(self, current_data: Dict[str, Any], 
                              film_metadata: Dict[str, Any]) -> Dict[str, Any]:
        """
        Generate comprehensive opening weekend prediction with uncertainty.
        
        Combines multiple models and provides scenario analysis.
        """
        if self.primary_model is None:
            raise ValueError("Model must be trained before making predictions")
        
        # Prepare feature vector from current data
        features = self._prepare_prediction_features(current_data, film_metadata)
        X_pred = np.array([features[col] for col in self.all_features if col in features])
        X_pred_scaled = self.scaler.transform(X_pred.reshape(1, -1))
        
        # Primary model prediction
        base_prediction = self.primary_model.predict(X_pred_scaled)[0]
        
        # Ensemble predictions for uncertainty
        ensemble_preds = self._get_ensemble_predictions(X_pred_scaled)
        
        # Uncertainty quantification
        uncertainty = self._calculate_prediction_uncertainty(base_prediction, ensemble_preds)
        
        # Scenario analysis
        scenarios = self._generate_scenarios(X_pred_scaled, film_metadata)
        
        # Risk assessment
        risks = self._assess_prediction_risks(features, uncertainty)
        
        # Explanation and feature contributions
        explanation = self._explain_prediction(features, base_prediction)
        
        return {
            'base_prediction': round(base_prediction / 1_000_000, 1),  # Convert to $M
            'confidence_interval': uncertainty,
            'ensemble_predictions': {
                'mean': np.mean(ensemble_preds),
                'std': np.std(ensemble_preds),
                'range': [np.min(ensemble_preds), np.max(ensemble_preds)]
            },
            'scenarios': scenarios,
            'risk_assessment': risks,
            'explanation': explanation,
            'prediction_timestamp': datetime.now().isoformat(),
            'model_version': 'v1.0-ensemble'
        }
    
    def _prepare_prediction_features(self, current_data: Dict, 
                                   film_metadata: Dict) -> Dict[str, float]:
        """Prepare complete feature vector for prediction."""
        features = {}
        
        # Performance indicators (current data)
        performance_keys = {
            'friday_gross': current_data.get('daily_gross', 0),
            'theater_count': current_data.get('theater_count', 0),
            'per_theater_avg': current_data.get('per_theater_average', 0),
            'online_sales_ratio': current_data.get('online_sales_ratio', 0.5)
        }
        features.update(performance_keys)
        
        # Film metadata features
        film_keys = {
            'production_budget': film_metadata.get('production_budget', 0),
            'marketing_budget': film_metadata.get('marketing_budget', 0),
            'genre_primary': self._encode_categorical(film_metadata.get('primary_genre', ''), 'genre_primary'),
            'mpaa_rating': self._encode_categorical(film_metadata.get('mpaa_rating', ''), 'mpaa_rating'),
            'runtime_minutes': film_metadata.get('runtime', 120),
            'franchise_flag': 1 if film_metadata.get('is_franchise', False) else 0
        }
        features.update(film_keys)
        
        # Fill missing features with reasonable defaults
        for feature in self.all_features:
            if feature not in features:
                features[feature] = self._get_default_value(feature)
        
        # Apply any necessary transformations
        features = self._transform_features(features)
        
        return features
    
    def _get_ensemble_predictions(self, X_scaled: np.ndarray) -> List[float]:
        """Generate predictions from ensemble models for uncertainty estimation."""
        predictions = []
        
        # Primary model
        if self.primary_model:
            predictions.append(self.primary_model.predict(X_scaled)[0])
        
        # Add predictions from secondary models if available
        for model_name, model in self.ensemble_models.items():
            try:
                pred = model.predict(X_scaled)[0]
                predictions.append(pred)
            except:
                continue
        
        return predictions if predictions else [0.0]
